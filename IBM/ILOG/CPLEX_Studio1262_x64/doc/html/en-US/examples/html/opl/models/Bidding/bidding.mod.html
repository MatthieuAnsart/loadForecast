012: /*******************************************************************
013:  OPL Model for CONTRACT BIDDING 
014: 
015: This model is described in the documentation. 
016: See IDE and OPL > Language and Interfaces Examples.
017:  
018:  *******************************************************************/
019: 
020: 
021: // DATA
022: // Custom data structures
023: tuple task {
024:   key string  bid;
025:   key string  task;
026:   float   price;
027:   int     option;
028: }
029: 
030: tuple bid {
031:   string  bid;
032:   string  task;
033: }
034: 
035: 
036: tuple penalty {
037:    key string task;
038:    float  cost;
039: }
040: 
041: // Basic data
042: {task}   TaskSet = ...;    // Bids for tasks
043: {string}    Tasks=... ;       // All tasks
044: {penalty}   TaskPenalty=...;  // Penalty for unassigned tasks. 
045:                               //If cost is -1, it means it must be assigned.
046: 
047: // Inferred sets
048: 
049:   // Set of <bids, task> pairs; 
050:   // If task ="REQD", it means that all tasks are mandatory for the bid; 
051:   // if task=a task name, it means the task is optional.
052: {bid}    Bids = {<t.bid, "REQD"> | t in TaskSet : t.option == 0} union 
053:                    {<t.bid, t.task> | t in TaskSet : t.option == 1};
054: 
055:                      
056:   // Set of all bids including task u; "REQD" indicates u is mandatory for the bid
057: {bid}    BidTasks[u in Tasks] = {<t.bid, "REQD"> | t in TaskSet : t.option == 0 && t.task == u} union 
058:                                    {<t.bid, t.task> | t in TaskSet : t.option == 1 && t.task == u};
059: 
060: // Inferred data
061: 
062: float BidCost[<b,t> in Bids] =
063:    sum (<b,t,price,opt> in TaskSet) price +
064:    sum (<b,t2,price,opt> in TaskSet : t == "REQD" && opt == 0) price;
065: 
066: 
067: assert sum (b in Bids) BidCost[b] >= 0;
068: 
069: 
070: float MaxBidCost = max (b in Bids) BidCost[b];
071: float Penalty[Tasks] = 
072:   [ t.task: (t.cost==-1) ? 1+MaxBidCost : t.cost | t in TaskPenalty];
073: 
074: 
075: // Decision variables
076: dvar boolean Award[Bids]; // 1 if the bid is awarded; 0 otherwise
077: dvar boolean Avoid[Tasks]; // 1 if the task is not assigned; 0 otherwise
078: 
079: // Objective value = cost of accepted bids + penalty for unassigned tasks
080: dexpr float AcceptedBidsCost =
081:   sum (b in Bids) BidCost[b]*Award[b];
082: dexpr float UnassignedTaskPenaltyCost = 
083:   sum (t in Tasks) Penalty[t]*Avoid[t];
084: 
085: 
086: // MODEL
087: minimize AcceptedBidsCost +UnassignedTaskPenaltyCost;
088: subject to {    
089:   // Assignment constraint: each task must be assigned to a bid
090:   forall (t in Tasks)
091:     ctAssign: sum (b in BidTasks[t]) Award[b] == 1 - Avoid[t];
092:        
093:   
094: }
095: 
096: tuple AwardSolutionT{ 
097: bid Bids;       int value; 
098: };
099: {AwardSolutionT} AwardSolution = {<i0,Award[i0]> | i0 in Bids};
100: tuple AvoidSolutionT{ 
101:         string Tasks; 
102:         int value; 
103: };
104: {AvoidSolutionT} AvoidSolution = {<i0,Avoid[i0]> | i0 in Tasks};
105: 
106: 
107: // POST-PROCESSING
108: {bid} Awarded = {<t.bid, t.task> | t in TaskSet :
109:   (t.option == 0 && Award[<t.bid, "REQD">] == 1) ||
110:   (t.option == 1 && Award[<t.bid, t.task>] == 1)};
111: {string} Avoided = {t | t in Tasks : Avoid[t] == 1};
112: int TotalAvoided = sum (t in Tasks) Avoid[t];
113: 
114: execute DISPLAY {
115:    writeln("Award = ", Award);
116:    writeln("Awarded = ", Awarded);
117:    writeln("Avoided = ", Avoided);
118:    writeln("TotalAvoided = ", TotalAvoided);
119:    for (var b in Bids) {
120:       writeln(b,  " ", BidCost[b]);
121:    }
122: }
