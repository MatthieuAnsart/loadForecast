12: /* ------------------------------------------------------------
13: 
14: Problem Description
15: -------------------
16: 
17: The problem involves choosing colors for the countries on a map in 
18: such a way that at most four colors (blue, white, yellow, green) are 
19: used and no neighboring countries are the same color. In this exercise, 
20: you will find a solution for a map coloring problem with six countries: 
21: Belgium, Denmark, France, Germany, Luxembourg, and the Netherlands. 
22: 
23: ------------------------------------------------------------ */
24: using CP;
25: 
26: range r = 0..3;
27: 
28: string Names[r] = ["blue", "white", "yellow", "green"]; 
29: 
30: dvar int Belgium in r;
31: dvar int Denmark in r;
32: dvar int France in r;
33: dvar int Germany  in r;
34: dvar int Luxembourg in r;
35: dvar int Netherlands in r;
36: 
37: subject to {  
38:    Belgium != France; 
39:    Belgium != Germany; 
40:    Belgium != Netherlands;
41:    Belgium != Luxembourg;
42:    Denmark != Germany; 
43:    France != Germany; 
44:    France != Luxembourg; 
45:    Germany != Luxembourg;
46:    Germany != Netherlands;    
47: }
48: 
49: execute {
50:    writeln("Belgium:     ", Names[Belgium]);
51:    writeln("Denmark:     ", Names[Denmark]);
52:    writeln("France:      ", Names[France]);
53:    writeln("Germany:     ", Names[Germany]);
54:    writeln("Luxembourg:  ", Names[Luxembourg]);
55:    writeln("Netherlands: ", Names[Netherlands]);
56: }
57: 
58: tuple resultT {
59:         string name;
60:         string value;
61: };
62: {resultT} solution = {};
63: execute{
64:    solution.add("Belgium", Names[Belgium]);
65:    solution.add("Denmark", Names[Denmark]);
66:    solution.add("France", Names[France]);
67:    solution.add("Germany", Names[Germany]);
68:    solution.add("Luxembourg", Names[Luxembourg]);
69:    solution.add("Netherlands", Names[Netherlands]);
70:    writeln(solution);
71: }
