011: 
012:  include "FlightPairing_data.mod";
013:  include "PairingGeneration_data.mod";
014:  
015:  using CP;
016:  
017:  int NbFlights  = card(flights);
018:  //int NbAirports = card(airports);
019:  int MaxLegs = parameters.maxNbLegs;
020:  int latestArrivalTime = max(f in flights) f.arrivalTime;
021:  int MaxWorkTime = parameters.workDurationMax;
022:  
023:  range FlightRange  = 1..NbFlights;
024:  range LegRange     = 1..MaxLegs;
025:  
026:  // time horizon is limited by latest arrival time.
027:  range horizon      = 0..latestArrivalTime;
028:  
029:  /// intermediate arrays for CP
030:  TFlightData FlightsByIndex[ f in FlightRange ] = item(flights,f-1);
031:  int arrivalTimes  [ f in FlightRange ] = FlightsByIndex[f].arrivalTime;
032:  int departureTimes[ f in FlightRange ] = FlightsByIndex[f].departureTime;
033:  int FlyingTimes[ f in FlightRange ] = FlightsByIndex[f].arrivalTime-FlightsByIndex[f].departureTime;
034:  
035:  int fromAirportIndex[ f in FlightRange] = ord(airports,<FlightsByIndex[f].fromAirportName>);
036:  int toAirportIndex  [ f in FlightRange] = ord(airports,<FlightsByIndex[f].toAirportName>);
037:  
038:  TAirport originAirport     [f in FlightRange] = item( airports, ord(airports,<FlightsByIndex[f].fromAirportName>) );
039:  //TAirport destinationAirport[f in FlightRange] = item( airports, ord(airports,<FlightsByIndex[f].originAirportName>) );
040:   
041:  int originTransitTimeMin[ f in FlightRange ] = originAirport[f].transitTimeMin;
042:  int originTransitTimeMax[ f in FlightRange ] = originAirport[f].transitTimeMax;
043:  
044:  /// external generation parameters
045: 
046:  {int} forcedCoveredFlights = ...;
047:  int costMax = ...;
048:  int tabuLength = ...; // crude way to avoid finding the same
049:  
050:  {TPairingLeg} genLegs;
051:  TPairingCost genCost;
052:  
053:  /**
054:   ** Pairings are modeled by a sequence of variables ranging over flights.
055:   ** 
056:   */
057:  dvar int flightVars[ LegRange ] in FlightRange;
058:  
059:  dvar boolean isActualLeg[ LegRange];
060:  
061:  dvar int+ startTimeVar in horizon;
062:  dvar int+ endTimeVar   in horizon;
063:  dvar int+ totalFlyingTime in 0..parameters.flyingDurationMax;
064:  
065:  /// pay
066:  dvar int+ costVar in parameters.minPay..999999999;
067:  dvar int+ workTimeCostVar;
068:  dvar int+ flyingTimeCostVar;
069:  
070:  dexpr int pairingLengthExp = sum(l in LegRange) isActualLeg[l];
071:  
072:  dvar int+ costVars[1..3];
073:  
074:  execute DEBUG {
075:    if ( PairingGenerationData.DebugMode >= 1 ) {
076:      writeln("-- searching for pairings with cost < ", costMax, ", length !=", tabuLength, ", forced=", forcedCoveredFlights); 
077:     }
078:  }
079:  
080: //minimize costVar;
081: subject to
082:  {
083:     // forbid one-leg pairings
084:     flightVars[1] != flightVars[2];
085:     isActualLeg[1] == 1;
086:     pairingLengthExp != tabuLength;
087:     pairingLengthExp >= 4;
088:     
089:     // relate status vars to flight vars
090:     forall(l in LegRange: l > 1) {
091:       isActualLeg[l] == (flightVars[l] != flightVars[l-1]);    
092:     }
093:     
094:     // return home at the end of the chain
095:     fromAirportIndex[ flightVars[1] ] == toAirportIndex[ flightVars[ MaxLegs]];
096:     
097:     // either the chain ends or the airport chain is continuous
098:     forall(l in 2..MaxLegs) {
099:        (0 == isActualLeg[l])
100:        ||
101:        (fromAirportIndex[ flightVars[l] ] == toAirportIndex[ flightVars[l-1]]);
102:     }
103:   
104:     forall (l in 2..MaxLegs) {
105:        (0 == isActualLeg[l])    
106:      ||
107:      (
108:       departureTimes[ flightVars[l] ] >= arrivalTimes[ flightVars[l-1]]
109:       + originTransitTimeMin[ flightVars[l] ]
110:       );
111:      }
112:     forall (l in 2..MaxLegs) {       
113:         (isActualLeg[l]==0)
114:       || 
115:       (
116:       departureTimes[ flightVars[l] ] <= arrivalTimes[ flightVars[l-1]]
117:       + originTransitTimeMax[ flightVars[l] ]
118:       );
119:     }
120:    
121: 
122:     
123:     /// propagate end of chain to the right
124:     forall (l in LegRange: 1 < l) {
125:        isActualLeg[l-1] >= isActualLeg[l];
126:     }
127:     
128:     // total flying time is counted only on "actual" legs
129:     totalFlyingTime == 
130:     FlyingTimes[ flightVars[1]]
131:     + sum(l in 2..MaxLegs) ( FlyingTimes[ flightVars[l] ] * isActualLeg[l]);
132:     
133:     startTimeVar == departureTimes[ flightVars[1]];
134:     endTimeVar   == arrivalTimes  [ flightVars[MaxLegs] ];
135:     
136:     (endTimeVar <= startTimeVar + MaxWorkTime);
137:     
138:     flyingTimeCostVar >= parameters.flyingPayRate * totalFlyingTime;
139:     flyingTimeCostVar <= parameters.flyingPayRate * totalFlyingTime+1;
140:     workTimeCostVar >= parameters.workPayRate * (endTimeVar - startTimeVar);
141:     workTimeCostVar <= parameters.workPayRate * (endTimeVar - startTimeVar)+1;
142:     
143:     /// all that pain to use max[]
144:     costVars[1] == parameters.minPay;
145:     costVars[2] == workTimeCostVar;
146:     costVars[3] == flyingTimeCostVar;
147:     
148:     costVar == max(j in 1..3) costVars[j];
149:     
150:      // at least one forced flight is covered.
151:      if ( card(forcedCoveredFlights) > 0)
152:        sum (f in forcedCoveredFlights) count(flightVars, f ) >= 1;
153:     
154:     // ceiling on cost.
155:     costVar < costMax;
156:  }
157:  
158:  execute STORE {
159:    var lastTime = -1;
160:    var lastFlight = null;
161:    genLegs.clear();
162:    for (var l in LegRange) {
163:      var flightRank = flightVars[l];
164:      var flight = FlightsByIndex[flightRank];
165:      genLegs.add(1, l, flightRank);
166:      
167:      if ( flight == lastFlight ) break;
168:      //writeln("* leg #", l, ", f#", flightRank, flight.name, ", from ", flight.fromAirportName, " to ", flight.toAirportName);
169:      var idle = lastTime>=0? flight.departureTime - lastTime: 0;
170:      //writeln("* idle: ", idle, ", fly time=", flight.arrivalTime-flight.departureTime);
171:      lastTime = flight.arrivalTime;
172:      lastFlight = flight;
173:    }
174:    genCost.id = 1;
175:    genCost.cost = costVar;
176:    //writeln(" work=", (endTimeVar - startTimeVar), ", flying=", totalFlyingTime);
177:    //writeln(" cost=", costVar, ", work cost=", workTimeCostVar, ", fly cost=", flyingTimeCostVar);
178:  }
