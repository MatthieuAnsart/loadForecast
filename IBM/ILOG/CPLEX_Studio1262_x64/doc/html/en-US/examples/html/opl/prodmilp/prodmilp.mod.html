12: {string} Products = ...;
13: {string} Resources = ...;
14: {string} Machines = ...;
15: float MaxProduction = ...;
16: 
17: tuple typeProductData {
18:   float demand;
19:   float incost;
20:   float outcost;
21:   float use[Resources];
22:   string machine;
23: }
24: 
25: typeProductData Product[Products] = ...;
26: float Capacity[Resources] = ...;
27: float RentCost[Machines] = ...;
28: 
29: dvar boolean Rent[Machines];
30: dvar float+ Inside[Products];
31: dvar float+ Outside[Products];
32: 
33: 
34: minimize
35:   sum( p in Products ) 
36:     ( Product[p].incost * Inside[p] + 
37:       Product[p].outcost * Outside[p] ) +
38:   sum( m in Machines ) 
39:     RentCost[m] * Rent[m];
40:    
41: subject to {
42:   forall( r in Resources )
43:     ctCapacity:
44:       sum( p in Products ) 
45:         Product[p].use[r] * Inside[p] <= Capacity[r];
46: 
47:   forall( p in Products )
48:     ctDemand: 
49:       Inside[p] + Outside[p] >= Product[p].demand;
50: 
51:   forall( p in Products )
52:     ctMaxProd:
53:       Inside[p] <= MaxProduction * Rent[Product[p].machine];
54: }
55:

57:         string Products; 
58:         float value; 
59: };
60: {InsideSolutionT} InsideSolution = {<i0,Inside[i0]> | i0 in Products};
61: tuple OutsideSolutionT{

63:         float value; 
64: };
65: {OutsideSolutionT} OutsideSolution = {<i0,Outside[i0]> | i0 in Products};
66: tuple RentSolutionT{

68:         int value; 
69: };
70: {RentSolutionT} RentSolution = {<i0,Rent[i0]> | i0 in Machines};
