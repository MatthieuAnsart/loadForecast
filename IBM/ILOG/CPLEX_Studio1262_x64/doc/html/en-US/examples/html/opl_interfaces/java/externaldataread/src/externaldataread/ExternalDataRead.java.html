14: // This is a very simple text reader used to demonstrate how
15: // OPL can use an externally defined data source.
16: // See documentation on externaldataread project
17: 
18:  public class ExternalDataRead extends IloCustomOplDataSource
19: {
20:         public ExternalDataRead(IloOplModel oplModel)
21:         {
22:                 super(IloOplFactory.getOplFactoryFrom(oplModel));
23:         }
24:         public void customRead()
25:         {
26:                 IloOplDataHandler handler = getDataHandler();
27:                 handler.startElement("a");
28:                 handler.addIntItem(1);
29:                 handler.endElement();
30:         }
31: }
