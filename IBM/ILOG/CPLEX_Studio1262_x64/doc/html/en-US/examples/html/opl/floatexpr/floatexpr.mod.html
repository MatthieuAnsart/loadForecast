12: using CP;
13: 
14: dvar int i in 1..100;
15: dvar int j in 1..100;
16: dvar int k in 1..99;
17: 
18: dexpr float kf = k/100;
19: dexpr float ar[i in 1..10] = k/i;
20: 
21: minimize kf + sum(i in 1..10)ar[i];
22: 
23: subject to {
24:    i*(1+kf) == j;
25: }
26: 
27: tuple solutionT{
28:         int i;
29:         int j;
30:         int k;
31: };
32: 
33: {solutionT} solution = {};
34: execute{
35: solution.add(i,j,k);
36: writeln(solution);
