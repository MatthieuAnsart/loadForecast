12: using CP;
13: 
14: int scale=8;
15: range ChessBoard = 1..scale*scale;
16: 
17: 
18: tuple coord
19: {
20:  int x;
21:  int y;  
22: };
23: 
24: {coord} Knightmoves2D[i in 1..scale][j in 1..scale]=
25: {<i+2,j+1>,<i+2,j-1>,<i+1,j+2>,<i+1,j-2>,
26: <i-2,j+1>,<i-2,j-1>,<i-1,j+2>,<i-1,j-2>}
27: inter {<x,y> | x in 1..scale ,y in 1..scale};
28: 
29: {int} Knightmove[i in ChessBoard]={(x-1)*scale+y | <x,y> in Knightmoves2D[(i-1) div scale+1][(i-1) % scale+1]};
30: 
31: dvar int jump[ChessBoard] in ChessBoard;
32: dvar int Sequence[ChessBoard] in ChessBoard;
33: 
34: execute {
35:                 var f = cp.factory;
36:         cp.setSearchPhases(f.searchPhase(Sequence));
37:         cp.param.defaultInferenceLevel="Extended";
38: }
39: 
40: subject to {
41:    forall(p in ChessBoard)
42:       jump[p] in Knightmove[p];
43: 
44:     Sequence[1] == jump[1];
45:     forall(p in 2..scale*scale)
46:       Sequence[p] == jump[Sequence[p-1]];
47: 
48:    allDifferent(Sequence);
49:    allDifferent(jump);
50:    
51:    Sequence[scale*scale] == 1;
52: 
53: 
54:    forall(p in ChessBoard)
55:       sum(c in Knightmove[p]) (jump[c] == p) == 1;  
56: 
57: 
58: };
59: 
60: 
61: int rank[1..scale,1..scale]=[ (Sequence[i]-1) % scale+1 : [ (Sequence[i]-1) div scale+1 : i ] | i in 1..scale *scale]; 
62: 
63: execute
64: {
65:  // rank is the human oriented result
66:  writeln(rank);  
67: }
68: 
69: 
70: tuple SequenceSolutionT{ 
71:         int ChessBoard; 
72:         int value; 
73: };
74: {SequenceSolutionT} SequenceSolution = {<i0,Sequence[i0]> | i0 in ChessBoard};
75: tuple jumpSolutionT{ 
76:         int ChessBoard; 
77:         int value; 
78: };
79: {jumpSolutionT} jumpSolution = {<i0,jump[i0]> | i0 in ChessBoard};
