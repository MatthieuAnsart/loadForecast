012: /******************************************************************************
013:  * 
014:  * OPL Model for Airline Yield Management
015:  * Stochastic formulation
016:  * 
017:  *
018:  * This model implements the example of Yield Management 
019:  * from "Model Building in Mathematical Programming"
020:  * by H. Paul Williams 
021:  *
022: ******************************************************************************/
023: 
024: // Data
025: //=======
026: 
027: // Seat classes available in the plane
028: // 10% of seats from a class can be transferred to an adjacent class.
029: {string} classes = {"First", "Business", "Economy"};
030: int capacity[classes] = [37, 38, 47];
031: 
032: // Planned periods 
033: range periods = 1..3;
034: 
035: // 3 scenarios are studied with their corresponding probability
036: range scenarios = 1..3;
037: float probaScenarios[scenarios] = [0.1, 0.7, 0.2];
038: 
039: // 3 pricing options for each class and period
040: range options = 1..3;
041: int priceOptions[periods][classes][options] = ...;
042: 
043: // Demands given by analytics for each period, scenario, class and price option
044: int demand[periods][scenarios][classes][options] = ...;
045: 
046: int planeBookPrice = ...;
047: 
048: // Decision variables
049: //=====================
050: 
051: // Boolean variables for each period and each class on selected price option
052: dvar boolean PriceLevel1[classes][options];
053: dvar boolean PriceLevel2[scenarios][classes][options];
054: dvar boolean PriceLevel3[scenarios][scenarios][classes][options];
055: 
056: // Number of seats 
057: dvar float+ Seats1[scenarios][classes][options]; // seats sold in wk 1 for scenario s, class c, price option o 
058: dvar float+ Seats2[scenarios][scenarios][classes][options]; // seats sold in wk 2 for scenario s1 for wk1, s2 for wk2, class c, price option o
059: dvar float+ Seats3[scenarios][scenarios][scenarios][classes][options]; //seats sold in wk 3 for scenario s1 for wk1, s2 for wk2, s3 for wk3, class c, price option o 
060: 
061: // Slack and Excess of capacity per class
062: dvar int+ U[scenarios][scenarios][scenarios][classes];
063: dvar int+ V[scenarios][scenarios][scenarios][classes];
064: 
065: // Number of planes to book (up to 6)
066: dvar int NbPlanes in 1..6; 
067: 
068: dvar int Revenue1[s1 in scenarios][c in classes][o in options] ;
069: dvar int Revenue2[s1 in scenarios][s2 in scenarios][c in classes][o in options] ;
070: dvar int Revenue3[s1 in scenarios][s2 in scenarios][s3 in scenarios][c in classes][o in options];
071: 
072:   
073: // Objective
074: //============ 
075: dexpr float revenue1 = sum (s1 in scenarios,c in classes, o in options) probaScenarios[s1]*Revenue1[s1][c][o];
076: dexpr float revenue2 = sum(s1,s2 in scenarios,c in classes, o in options) probaScenarios[s1]*probaScenarios[s2]*Revenue2[s1][s2][c][o];
077: dexpr float revenue3 = sum(s1,s2,s3 in scenarios,c in classes, o in options) probaScenarios[s1]*probaScenarios[s2]*probaScenarios[s3]*Revenue3[s1][s2][s3][c][o];
078: dexpr float planeCost = planeBookPrice * NbPlanes;
079: 
080: dexpr float totalObj = revenue1 + revenue2 + revenue3 - planeCost;
081:  
082: maximize revenue1 + revenue2 + revenue3 - planeCost;
083: 
084: subject to {
085:    // linearisation
086:    forall (s1 in scenarios, c in classes, o in options) {
087:       Revenue1[s1][c][o] - priceOptions[1][c][o]*Seats1[s1][c][o] <= 0;
088:       priceOptions[1][c][o]*Seats1[s1][c][o] - Revenue1[s1][c][o] + priceOptions[1][c][o]*demand[1][s1][c][o]*PriceLevel1[c][o] <= priceOptions[1][c][o]*demand[1][s1][c][o]; 
089:    }
090:    forall (s1,s2 in scenarios, c in classes, o in options) {
091:       Revenue2[s1][s2][c][o] - priceOptions[2][c][o]*Seats2[s1][s2][c][o] <= 0;
092:       priceOptions[2][c][o]*Seats2[s1][s2][c][o] - Revenue2[s1][s2][c][o] + priceOptions[2][c][o]*demand[2][s2][c][o]*PriceLevel2[s1][c][o] <= priceOptions[2][c][o]*demand[2][s2][c][o]; 
093:    }
094:    forall (s1,s2,s3 in scenarios, c in classes, o in options) {
095:       Revenue3[s1][s2][s3][c][o] - priceOptions[3][c][o]*Seats3[s1][s2][s3][c][o] <= 0;
096:       priceOptions[3][c][o]*Seats3[s1][s2][s3][c][o] - Revenue3[s1][s2][s3][c][o] + priceOptions[3][c][o]*demand[3][s3][c][o]*PriceLevel3[s1][s2][c][o] <= priceOptions[3][c][o]*demand[3][s3][c][o]; 
097:    }
098: 
099:    // At each period, and for each class, only one price option is selected.
100:    forall(c in classes) 
101:       sum (o in options) PriceLevel1[c][o] == 1;
102:    forall (c in classes, s1 in scenarios) 
103:       sum (o in options) PriceLevel2[s1][c][o] == 1;
104:    forall (c in classes, s1,s2 in scenarios) 
105:       sum (o in options) PriceLevel3[s1][s2][c][o] == 1;
106:  
107:    // Seat Capacity  
108:    forall(s1,s2,s3 in scenarios, c in classes) 
109:       ctSeatCapacities: 
110:          sum (o in options) (Seats1[s1][c][o] + Seats2[s1][s2][c][o] + Seats3[s1][s2][s3][c][o]) 
111:          <= capacity[c]*NbPlanes + U[s1][s2][s3][c] - V[s1][s2][s3][c];  
112:    forall(s1,s2,s3 in scenarios, c in classes) {
113:       U[s1][s2][s3][c] <= 0.1*capacity[c];
114:       V[s1][s2][s3][c] <= 0.1*capacity[c];
115:    }
116:    forall(s1,s2,s3 in scenarios)
117:      sum (c in classes) (U[s1][s2][s3][c] - V[s1][s2][s3][c]) == 0;
118:      
119:    // Number of seats sold is lower than demand at each period    
120:    forall (s1 in scenarios, c in classes, o in options)
121:       ctDemand1: Seats1[s1][c][o] <= demand[1][s1][c][o]*PriceLevel1[c][o];    
122:    forall (s1,s2 in scenarios, c in classes, o in options)
123:       ctDemand2: Seats2[s1][s2][c][o] <= demand[2][s2][c][o]*PriceLevel2[s1][c][o];  
124:    forall (s1,s2,s3 in scenarios, c in classes, o in options)
125:       ctDemand3: Seats3[s1][s2][s3][c][o] <= demand[3][s3][c][o]*PriceLevel3[s1][s2][c][o];  
126: 
127: };
128: 
129: 
130: tuple Revenue1SolutionT{ 
131:         int scenarios; 
132:         string classes; 
133:         int options; 
134:         int value; 
135: };
136: {Revenue1SolutionT} Revenue1Solution = {<i0,i1,i2,Revenue1[i0][i1][i2]> | i0 in scenarios,i1 in classes,i2 in options};
137: tuple Revenue2SolutionT{ 
138:         int scenarios1; 
139:         int scenarios2; 
140:         string classes; 
141:         int options; 
142:         int value; 
143: };
144: {Revenue2SolutionT} Revenue2Solution = {<i0,i1,i2,i3,Revenue2[i0][i1][i2][i3]> | i0 in scenarios,i1 in scenarios,i2 in classes,i3 in options};
145: tuple Revenue3SolutionT{ 
146:         int scenarios1; 
147:         int scenarios2; 
148:         int scenarios3; 
149:         string classes; 
150:         int options; 
151:         int value; 
152: };
153: {Revenue3SolutionT} Revenue3Solution = {<i0,i1,i2,i3,i4,Revenue3[i0][i1][i2][i3][i4]> | i0 in scenarios,i1 in scenarios,i2 in scenarios,i3 in classes,i4 in options};
154: tuple Seats1SolutionT{ 
155:         int scenarios; 
156:         string classes; 
157:         int options; 
158:         float value; 
159: };
160: {Seats1SolutionT} Seats1Solution = {<i0,i1,i2,Seats1[i0][i1][i2]> | i0 in scenarios,i1 in classes,i2 in options};
161: tuple PriceLevel1SolutionT{ 
162:         string classes; 
163:         int options; 
164:         int value; 
165: };
166: {PriceLevel1SolutionT} PriceLevel1Solution = {<i0,i1,PriceLevel1[i0][i1]> | i0 in classes,i1 in options};
167: tuple Seats2SolutionT{ 
168:         int scenarios1; 
169:         int scenarios2; 
170:         string classes; 
171:         int options; 
172:         float value; 
173: };
174: {Seats2SolutionT} Seats2Solution = {<i0,i1,i2,i3,Seats2[i0][i1][i2][i3]> | i0 in scenarios,i1 in scenarios,i2 in classes,i3 in options};
175: tuple PriceLevel2SolutionT{ 
176:         int scenarios; 
177:         string classes; 
178:         int options; 
179:         int value; 
180: };
181: {PriceLevel2SolutionT} PriceLevel2Solution = {<i0,i1,i2,PriceLevel2[i0][i1][i2]> | i0 in scenarios,i1 in classes,i2 in options};
182: tuple Seats3SolutionT{ 
183:         int scenarios1; 
184:         int scenarios2; 
185:         int scenarios3; 
186:         string classes; 
187:         int options; 
188:         float value; 
189: };
190: {Seats3SolutionT} Seats3Solution = {<i0,i1,i2,i3,i4,Seats3[i0][i1][i2][i3][i4]> | i0 in scenarios,i1 in scenarios,i2 in scenarios,i3 in classes,i4 in options};
191: tuple PriceLevel3SolutionT{ 
192:         int scenarios1; 
193:         int scenarios2; 
194:         string classes; 
195:         int options; 
196:         int value; 
197: };
198: {PriceLevel3SolutionT} PriceLevel3Solution = {<i0,i1,i2,i3,PriceLevel3[i0][i1][i2][i3]> | i0 in scenarios,i1 in scenarios,i2 in classes,i3 in options};
199: tuple USolutionT{ 
200:         int scenarios1; 
201:         int scenarios2; 
202:         int scenarios3; 
203:         string classes; 
204:         int value; 
205: };
206: {USolutionT} USolution = {<i0,i1,i2,i3,U[i0][i1][i2][i3]> | i0 in scenarios,i1 in scenarios,i2 in scenarios,i3 in classes};
207: tuple VSolutionT{ 
208:         int scenarios1; 
209:         int scenarios2; 
210:         int scenarios3; 
211:         string classes; 
212:         int value; 
213: };
214: {VSolutionT} VSolution = {<i0,i1,i2,i3,V[i0][i1][i2][i3]> | i0 in scenarios,i1 in scenarios,i2 in scenarios,i3 in classes};
215: 
216: 
217: execute DISPLAY_RESULTS {
218:   writeln("(Provisionally) book ", NbPlanes, " planes");
219:   writeln("expected profit = ", totalObj);  
220: };
