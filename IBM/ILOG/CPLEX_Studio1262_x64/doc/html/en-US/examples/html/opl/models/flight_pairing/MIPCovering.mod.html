12: /// --- 
13: /// This is the MIP variant of the set covering problem.
14: /// ---
15: include "FlightPairing_data.mod";
16: include "PairingGeneration_data.mod";
17: 
18: /// externals
19: {TPairingCost} pairingCosts = ...;
20: {TPairingLeg}  pairingLegs  = ...;
21: 
22: /// internal
23: int NbFlights = card(flights);
24: range FlightIndexRange = 1..NbFlights;
25: 
26: {int} pairing_ids = { pc.id | pc in pairingCosts };
27: 
28: /// locals
29: {int} covered[ p in pairing_ids] = { fr | <p, l, fr> in pairingLegs };
30: 
31: dvar boolean iUsageVars[ pairing_ids ];
32: 
33: dexpr float totalUsageCost = sum(pc in pairingCosts) (iUsageVars[pc.id] * pc.cost);
34: 
35: minimize totalUsageCost;
36: 
37: subject to { 
38:   forall (f in FlightIndexRange) {
39:     sum(p in pairing_ids: f in covered[p]) iUsageVars[p] >= 1;  
40:   }
41: }
42: 
43: int numberOfSelectedPairings = sum(p in pairing_ids) iUsageVars[p];
44: execute DEBUG {
45:   writeln("* MIP covering uses: ", numberOfSelectedPairings, " out of ", pairingLegs.size);
46: }
