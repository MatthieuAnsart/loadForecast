12: //
13: // This example is inspired from the talent hold cost scheduling problem
14: // described in:
15: //
16: // T.C.E Cheng, J. Diamond, B.M.T. Lin.  Optimal scheduling in film
17: // production to minimize talent holding cost.  Journal of Optimization
18: // Theory and Applications, 79:197-206, 1993.
19: //
20: // of which the 'Rehearsal problem' is a specific case:
21: //
22: // Barbara M. Smith.  Constraint Programming In Practice: Scheduling
23: //                    a Rehearsal.  Report APES-67-2003, September 2003.
24: // 
25: // See: http://www.csplib.org/Problems/prob039/
26: //

29: 
30: using CP;
31: 
32: execute{
33:         }
34: int numActors = ...;
35: range Actors = 1..numActors;
36: int actorPay[Actors] = ...;
37: int numScenes = ...;
38: range Scenes = 1..numScenes;
39: int sceneDuration[Scenes] = ...;
40: 
41: int actorInScene[Actors][Scenes]  = ...;
42: 
43: dvar int scene[Scenes] in Scenes;
44: dvar int slot[Scenes] in Scenes;
45: 
46: 
47: // First and last slots where each actor plays
48: dexpr int firstSlot[a in Actors] = min(s in Scenes:actorInScene[a][s] == 1) slot[s];
49: dexpr int lastSlot[a in Actors] = max(s in Scenes:actorInScene[a][s] == 1) slot[s];
50: 
51: // Expression for the waiting time for each actor
52: dexpr int actorWait[a in Actors] = sum(s in Scenes: actorInScene[a][s] == 0)  
53:    (sceneDuration[s] * (firstSlot[a] <= slot[s] && slot[s] <= lastSlot[a]));
54: 
55: // Expression representing the global cost
56: dexpr int idleCost = sum(a in Actors) actorPay[a] * actorWait[a];
57: 
58: minimize idleCost;
59: subject to {
60:    // use the slot-based secondary model
61:    inverse(scene, slot);
62: }
63: 
64: tuple slotSolutionT{ 
65:         int Scenes; 
66:         int value; 
67: };
68: {slotSolutionT} slotSolution = {<i0,slot[i0]> | i0 in Scenes};
69: tuple sceneSolutionT{ 
70:         int Scenes; 
71:         int value; 
72: };
73: {sceneSolutionT} sceneSolution = {<i0,scene[i0]> | i0 in Scenes};
