12: using CP;
13: 
14: int nbModel = ...;
15: int nbCard = ...;
16: int nbRack = ...;
17: 
18: range Models = 0..nbModel-1;
19: range Cards = 0..nbCard-1;
20: range Racks = 0..nbRack-1;
21: 
22: execute{
23:         }
24: 
25: 
26: tuple modelType {
27:    int power;
28:    int connectors;
29:    int price;
30: };
31: 
32: tuple cardType {
33:    int power;
34:    int quantity;
35: };
36: 
37: modelType model[Models] = ...;
38: cardType car[Cards] = ...;
39: 
40: int maxPrice = max(r in Models) model[r].price;
41: int maxCost = nbCard * maxPrice;
42: int powerData[i in Models] = model[i].power;
43: int connData[i in Models ] = model[i].connectors;
44: int priceData[i in Models ] = model[i].price;
45: 
46: dvar int rack[Racks] in Models;
47: dvar int counters[Racks][Cards] in 0..nbCard;
48: dvar int cost in 0..maxCost;
49: 
50: minimize
51:   cost;
52: subject to {
53:    forall(r in Racks)
54:       sum(c in Cards) counters[r][c] * car[c].power <= powerData[rack[r]];
55:    forall(r in Racks)
56:       sum(c in Cards) counters[r][c] <= connData[rack[r]];
57:    forall(c in Cards)
58:       sum(r in Racks) counters[r][c] == car[c].quantity;
59:    cost == sum(r in Racks) priceData[rack[r]];
60: };   
61: 
62: execute DISPLAY {
63:   writeln("cost = ", cost);
64:   writeln("rack = ", rack);
65: };
