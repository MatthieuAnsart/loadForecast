12: /**
13:  * Airport data. 
14:  * Each airport is identified by its three-letter id code
15:  * and a transit time range: each flight passing through the airport
16:  * must satisfy this transit range.
17:  */
18: tuple TAirport {
19:    key string id;
20:    int transitTimeMin;
21:    int transitTimeMax;
22: }
23: 
24: /**
25:  * Flight data;
26:  * A flight is identified by its id, its source and target airports
27:  * and time schedule.
28:  */
29: tuple TFlightData {
30:    key string name;
31:    string fromAirportName;
32:    string toAirportName;
33:    
34:    // times are in minutes...
35:    int departureTime;
36:    int arrivalTime;
37: }
38:  
39:  /**
40:   * Global parameters (one single tuple of this type)
41:   * maxmum work and fly time (regulations)
42:   * data pertaning to pay
43:   *  - minimum pay (flat sum)
44:   *  - work time pay rate per time unit
45:   *  - fly-time pay rate by time unit
46:   *
47:   * - maximum nb legs is 8 by default.
48:  */
49: tuple TParameters {
50:   int maxNbLegs;
51:   int workDurationMax;
52:   int flyingDurationMax;
53:   
54:   int minPay;
55:   float workPayRate;
56:   float flyingPayRate;
57: 
58: };
59:

61: {TAirport}    airports = ...;

63: 
64: TParameters parameters = ...;
