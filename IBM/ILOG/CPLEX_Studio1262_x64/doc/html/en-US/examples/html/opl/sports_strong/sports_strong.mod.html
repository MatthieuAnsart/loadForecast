012: using CP;
013: 
014: /* ------------------------------------------------------------
015: 
016: Problem Description
017: -------------------
018: 
019: The problem involves finding a schedule for a sports league. The league has 10 
020: teams that play games over a season of 18 weeks. Each team has a home arena and 
021: plays each other team twice during the season, once in its home arena and once in 
022: the opposing team's home arena. For each of these games, the team playing at its 
023: home arena is referred to as the home team; the team playing at the opponent's 
024: arena is called the away team. There are 90 games altogether.
025: 
026: Each of the 18 weeks in the season has five identical slots to which games can be 
027: assigned. Each team plays once a week. For each pair of teams, these two teams are 
028: opponents twice in a season; these two games must be scheduled in different halves 
029: of the season. Moreover, these two games must be scheduled at least six weeks 
030: apart. A team must play at home either the first or last week but not both.
031: 
032: A break is a sequence of consecutive weeks in which a team plays its games either 
033: all at home or all away. No team can have a break of three or more weeks in it. The
034: objective in this problem is to minimize the total number of breaks the teams play. 
035: 
036: ------------------------------------------------------------ */
037: 
038: int n = 10;
039: 
040: 
041: assert(n%2 == 0);
042: 
043: int nbWeeks = 2 * (n - 1);
044: int nbGamesPerWeek = n div 2;
045: int nbGames = n * (n - 1);
046: float mid = nbWeeks / 2 + 1;
047: int overlap = (n>=6) ? minl(n div 2, 6) : 0;
048: 
049: range R1 = 1..nbWeeks;
050: range R2 = 1..nbGamesPerWeek;
051: range R3 = 1..nbGames;
052: range R4 = 1..n;
053: 
054: dvar int games[R1][R2] in R3;
055: dvar int home[R1][R2] in R4;
056: dvar int away[R1][R2] in R4;
057: dvar int weekOfGame[R3] in  R1;
058: dvar int allSlots[R3] in R3;
059: dvar boolean playHome[R4][R1];
060: dvar int allGames[R3] = all[1..nbGames](w in R1, g in 1..nbGamesPerWeek) games[w][g];
061: dvar int teamBreaks[R4] in 0..nbWeeks div 2;
062: 
063: 
064: //
065: // For each play slot, set up correspondance between game id,
066: // home team, and away team
067: tuple PlaySlotTuple {
068:    int home;
069:    int away;
070:    key int gameId;
071: };
072: 
073: {PlaySlotTuple} playSlots = {<h, a, (h-1) * (n-1) + a - (a > h)>
074:                              | h, a in 1..n : a != h};
075: 
076: 
077: execute {
078:     var f = cp.factory;
079:     var phase = f.searchPhase(allGames, 
080:                               f.selectSmallest(f.varIndex(allGames)), 
081:                               f.selectRandomValue());
082:     cp.setSearchPhases(phase);
083:     cp.param.timeLimit=60;
084:     cp.param.logPeriod=10000;
085:     cp.param.DefaultInferenceLevel="Extended";
086: }
087: 
088: 
089: //
090: // Objective: minimize the number of `breaks'.  A break is
091: //            two consecutive home or away matches for a
092: //            particular team
093: dexpr int breakCount = sum(t in 1..n) teamBreaks[t];
094: 
095: dexpr int Games[w in 1..nbWeeks][g in 1..nbGamesPerWeek] =
096:   (home[w][g]-1)*(n-1)+away[w][g] - (away[w][g]>home[w][g]);
097: 
098: minimize breakCount;
099: subject to {
100:   // Calculate the unique game identifier for the home/away combination
101:   forall(w in 1..nbWeeks, g in 1..nbGamesPerWeek)
102:     games[w][g] == Games[w][g];
103: 
104:   // strengthen the constraints on these variables
105:   forall(w in 1..nbWeeks, g in 1..nbGamesPerWeek)
106:     strong(home[w][g], away[w][g], games[w][g]);
107: 
108:    // All teams play each week
109:    forall(w in 1..nbWeeks) {
110:      allDifferent(append(all(g in 1..nbGamesPerWeek) home[w][g], 
111:                          all(g in 1..nbGamesPerWeek) away[w][g])); 
112:    }
113: 
114:     //
115:     // Dual representation: for each game id, the play slot is maintained
116:     //
117:     inverse(all [1..nbGames](w in 1..nbWeeks, g in 1..nbGamesPerWeek) games[w][g], allSlots); 
118:     forall(g in 1..nbGames)
119:       weekOfGame[g] == ((allSlots[g]-1) div nbGamesPerWeek) + 1;
120: 
121: 
122:     //
123:     // Two half schedules.  Cannot play the same pair twice in the same half.
124:     // Plus, impose a minimum number of weeks between two games involving
125:     // the same teams (up to six weeks)
126:     //
127:     forall (<i,j,g1> in playSlots, <j,i,g2> in playSlots  : i < j) {
128:        (weekOfGame[g1] >= mid) == (weekOfGame[g2] < mid); 
129:        if (overlap != 0)
130:           abs(weekOfGame[g1] - weekOfGame[g2]) >= overlap;
131:     }
132:                
133:     //
134:     // Can't have three homes or three aways in a row.
135:     //
136:     forall (t in 1..n, w in 1..nbWeeks) {
137:        playHome[t][w] == count(all(g in 1..nbGamesPerWeek) home[w][g], t);
138:     }
139: 
140:     forall (t in 1..n, w in 1..nbWeeks - 2) {
141:        1 <= sum(k in w..w+2) playHome[t][k] <= 2;
142:     }
143:   
144:     //
145:     // If we start the season home, we finish away and vice versa.
146:     //
147:     forall(t in 1..n)
148:        teamBreaks[t] == sum(w in 2..nbWeeks) (playHome[t][w-1] == playHome[t][w]);
149:        
150:     forall (t in 1..n)
151:       playHome[t][1] != playHome[t][nbWeeks];
152: 
153: 
154:     //
155:     // Catalyzing constraints
156:     //
157: 
158:     // Each team plays home the same number of times as away
159:     forall (t in 1..n) {
160:        sum(w in 1..nbWeeks) playHome[t][w] == nbWeeks div 2;
161:     }
162: 
163:     // Breaks must be even for each team
164:     forall(t in 1..n)
165:        teamBreaks[t] % 2 == 0;
166: 
167:     //    
168:     // Symmetry breaking constraints
169:     // 
170:     // Teams are interchangeable.  Fix first week.
171:     // Also breaks reflection symmetry of the whole schedule.
172:     forall (g in 1..nbGamesPerWeek) {
173:        home[1][g] == g*2 - 1;
174:        away[1][g] == g*2;
175:     }
176: 
177:     // Order of games in each week in arbitrary.
178:     // Break symmetry by forcing an order.
179:     forall (w in 1..nbWeeks)
180:       forall(g in 2..nbGamesPerWeek)
181:         games[w][g] > games[w][g-1];
182: }
183: 
184: tuple gamesSolutionT{ 
185:         int R1; 
186:         int R2; 
187:         int value; 
188: };
189: {gamesSolutionT} gamesSolution = {<i0,i1,games[i0][i1]> | i0 in R1,i1 in R2};
190: tuple allGamesSolutionT{ 
191:         int R3; 
192:         int value; 
193: };
194: {allGamesSolutionT} allGamesSolution = {<i0,allGames[i0]> | i0 in R3};
195: tuple teamBreaksSolutionT{ 
196:         int R4; 
197:         int value; 
198: };
199: {teamBreaksSolutionT} teamBreaksSolution = {<i0,teamBreaks[i0]> | i0 in R4};
200: tuple homeSolutionT{ 
201:         int R1; 
202:         int R2; 
203:         int value; 
204: };
205: {homeSolutionT} homeSolution = {<i0,i1,home[i0][i1]> | i0 in R1,i1 in R2};
206: tuple awaySolutionT{ 
207:         int R1; 
208:         int R2; 
209:         int value; 
210: };
211: {awaySolutionT} awaySolution = {<i0,i1,away[i0][i1]> | i0 in R1,i1 in R2};
212: tuple allSlotsSolutionT{ 
213:         int R3; 
214:         int value; 
215: };
216: {allSlotsSolutionT} allSlotsSolution = {<i0,allSlots[i0]> | i0 in R3};
217: tuple weekOfGameSolutionT{ 
218:         int R3; 
219:         int value; 
220: };
221: {weekOfGameSolutionT} weekOfGameSolution = {<i0,weekOfGame[i0]> | i0 in R3};
222: tuple playHomeSolutionT{ 
223:         int R4; 
224:         int R1; 
225:         int value; 
226: };
227: {playHomeSolutionT} playHomeSolution = {<i0,i1,playHome[i0][i1]> | i0 in R4,i1 in R1};
228: 
229: 
230: 
231: int oponent[1..n][1..nbWeeks];
232: 
233: int breaks = sum(t in 1..n) teamBreaks[t];
234: execute {
235:    
236:       writeln("Solution at " + breaks);
237:       for (var j= 1; j <= nbWeeks; j++) {
238:          write("Week " + j + ": ");
239:          if (j < 10) write (" ");
240:          for (var i = 1; i <= nbGamesPerWeek; i++) {
241:             if (home[j][i] >= 10) 
242:               write(home[j][i]);
243:             else
244:               write(" " + home[j][i]);
245:             write("-");
246:             if (away[j][i] >= 10)
247:               write(away[j][i]);
248:             else
249:               write(away[j][i] + " ");
250:             write(" ");      
251:          }
252:          writeln();
253:       }
254:       writeln("Team schedules");
255:       for (i = 1; i <= n; i++) {
256:          write("T " + i + ":  ");
257:          var prev = -1;
258:          var brks = 0;
259:          for (j = 1; j <= nbWeeks; j++) {
260:             for (var k = 1; k <= nbGamesPerWeek; k++) {
261:                if (home[j][k] == i) {
262:                   oponent[i][j] = away[j][k];
263:                   if (away[j][k] >= 10)
264:                     write(away[j][k] + "H ")
265:                   else
266:                     write(" " + away[j][k] + "H ");
267:                   brks += (prev == 0);
268:                   prev = 0;    
269:                }
270:                if (away[j][k] == i) {
271:                   oponent[i][j] = home[j][k];
272:                   if (home[j][k] >= 10)
273:                     write(home[j][k] + "A ");
274:                   else
275:                     write(" " + home[j][k] + "A ");
276:                   brks += (prev == 1);
277:                   prev = 1;
278:                }
279:             }
280:          }
281:          writeln("  " + brks + " breaks");
282:       }
283:       writeln();
284: }
