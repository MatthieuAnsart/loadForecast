025: /*
026:   This model is described in the documentation. 
027: See IDE and OPL > Language and Interfaces Examples.
028:   
029:   Examples of the proper data format appear below.
030:   
031: Flight Leg:
032:    <5,   "ABZ", 1165, "SVG", 1376,144>
033:    
034:    Flt. ID = 5
035:    Origin    = "ABZ" (city code)
036:    Departure Time (in minutes) = 1165
037:    Destination = "SVG" (city cod)
038:    Arrival Time (in minutes) = 1376
039:    Aircraft Capacity = 144
040:    
041: Itinerary:
042:    < 1, 3, {0, 3, 4}, 30, 110>
043:    
044:    Itinerary Num. = 1
045:    Number of Flight Legs = 3
046:    Flight Leg ID #1 = 0
047:    Flight Leg ID #2 = 3
048:    Flight Leg ID #3 = 4
049:    Expected Demand = 30
050:    fare = 110   
051: */
052: 
053: // Flight Leg structure 
054: tuple flightLeg 
055: {
056:   key int        FltID;             //flight ID
057:   string     Origin;            //Originating Airport
058:   int        Dept;              //Departure Time
059:   string     Destination;       //Destination Airport
060:   int        Arrt;              //Arrival Time
061:   int        Cap;               //Capacity
062: }
063: 
064: // Set of Scheduled Flight Legs
065: {flightLeg} FlightLegs = ...; 
066: assert forall(fl in FlightLegs)  fl.Dept <= fl.Arrt;
067: 
068: {int} LegIDSet = {fl.FltID | fl in FlightLegs};
069: 
070: // Make sure there are no duplicate flight IDs!
071: assert card(LegIDSet) == card(FlightLegs);
072: 
073: flightLeg ArrayOfFlightLegs[LegIDSet] =
074:  [fl.FltID : fl | fl in FlightLegs];
075: 
076: 
077: // Maximum time that passenger is willing to wait between connecting flights
078: int MaxWaitTime = ...;
079: 
080: // Itinerary structure - Unlimited number of legs in itinerary
081: tuple itinerary
082: {
083:   key int ItinID;            //Itinerary ID
084:   int NumLegs;           //Number of legs in Itinerary    
085:   {int} LegIDs;          //Set of Legs in Itinerary
086:   float dmd;             //Expected Demand
087:   float fare;            //fare
088: }
089: 
090: // Set of desired Itineraries
091: {itinerary} Itineraries = ...; 
092: 
093: {int} ItinIDSet = {it.ItinID | it in Itineraries};
094: 
095: // Make sure there are no duplicate itinerary numbers!
096: assert card(ItinIDSet) == card(Itineraries);
097: 
098: int Maxis[i in Itineraries] = max(g in i.LegIDs) ArrayOfFlightLegs[g].Dept;
099: 
100: // An assertion ensuring that all itineraries are feasible             
101: assert
102:     forall(i in Itineraries, f1 in i.LegIDs: ArrayOfFlightLegs[f1].Dept < Maxis[i])
103:     1 ==
104:         sum(f2 in i.LegIDs:
105:              f1 != f2 &&
106:              ArrayOfFlightLegs[f2].Origin == ArrayOfFlightLegs[f1].Destination &&
107:              ArrayOfFlightLegs[f1].Arrt <= ArrayOfFlightLegs[f2].Dept &&
108:              ArrayOfFlightLegs[f2].Dept <= ArrayOfFlightLegs[f1].Arrt + MaxWaitTime
109: )
110:             1;
111: 
112: /******************************************************************************
113:  * MODEL DECLARATIONS
114:  ******************************************************************************/
115: 
116: dvar float+   Allocation[it in Itineraries] in 0..it.dmd;  // allocation level to itinerary        
117: 
118: /******************************************************************************
119:  * MODEL
120:  ******************************************************************************/
121: 
122: dexpr float TotalAllocation =
123:   sum(it in Itineraries) it.fare*Allocation[it];
124: 
125: maximize TotalAllocation;
126:     
127: subject to {
128:   forall(fl in FlightLegs) 
129:     ctCapLimit: 
130:       sum(it in Itineraries: fl.FltID in it.LegIDs) Allocation[it] <= fl.Cap;
131: }
132: 
133: 
134: tuple itineraryT{
135:   key int ItinID;        //Itinerary ID
136:   int NumLegs;           //Number of legs in Itinerary    
137:   float dmd;             //Expected Demand
138:   float fare;            //fare
139: }
140: tuple AllocationSolutionT{ 
141:         itineraryT Itineraries;         
142:         float value; 
143: };
144: {AllocationSolutionT} AllocationSolution = {<<i0.ItinID, i0.NumLegs, i0.dmd, i0.fare>,Allocation[i0]> | i0 in Itineraries};
145: execute{ 
146:         writeln(AllocationSolution);
147: }
148: 
149: execute DISPLAY {
150:    for(var fl in FlightLegs)
151:       if(ctCapLimit[fl].dual > 0.001)
152:          writeln("ctCapLimit[", fl, "].dual = ", ctCapLimit[fl].dual);
153: }
