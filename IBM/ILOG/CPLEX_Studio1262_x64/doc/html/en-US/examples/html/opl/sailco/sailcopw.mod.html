17: float ExtraCost = ...;

27:    sum(t in Periods)
28:        piecewise{ RegularCost -> Capacity ; ExtraCost } Boat[t] +
29:                   InventoryCost  * (sum(t in Periods) Inv[t]);

32:   ctInventory:

34:   forall(t in Periods)
35:     ctDemand:
36:       Boat[t] + Inv[t-1] == Inv[t] + Demand[t];
