012: include "FlightPairing_data.mod";
013: 
014:  /// this is actually a JS program
015:  
016:  // the maximum number of initial pairings to generate.
017:  int NumberOfInitialPairings = 3;
018:  
019:  
020: {TPairingLeg}  pairingLegs;
021: {TPairingCost} pairingCosts;
022: 
023:  main {
024:    includeScript("helpers.js");
025:    var status = 0;
026: 
027:    var MAX_INITIAL_PAIRINGS_PER_FLIGHT         = thisOplModel.NumberOfInitialPairings;   // Initial pairings to cover each flight
028:    var PrintPeriod  = 20;   // Print frequency
029:    
030:    var time  = 0;          // Procedure time
031:    var date0;
032:    var totalTime;            // Total computation time 
033: 
034:    /******************************************************************************
035:     * INITIALIZATION:
036:     * Generate initial pairings that cover all flights
037:     ******************************************************************************/
038:    
039:    // Load pairing cover model
040:    var generateOnePairingConfig =
041:       new IloOplRunConfiguration("GenerateOnePairing.mod", "FlightPairing.dat", "PairingGeneration.dat", "DefaultGenerateOnePairing.dat");
042:    var generatePairingModel = generateOnePairingConfig.oplModel;
043:    var generatePairingData  = generatePairingModel.dataElements;
044:    //generatePairingData.forcedCoveredFlight = -1;
045:    generatePairingModel.generate();
046:    
047:    var allFlights = generatePairingData.flights;
048:    var nbFlights = generatePairingData.flights.size;
049:    var maxNbLegs = generatePairingData.parameters.maxNbLegs;
050:    
051:    writeln("* starting CP search for initial pairings", ", #flights=", nbFlights, ", max. pairings/flight=", MAX_INITIAL_PAIRINGS_PER_FLIGHT);
052:    date0 = new Date();
053:    // Indices for master problem
054:    var legs  = thisOplModel.pairingLegs;
055:    var costs = thisOplModel.pairingCosts;
056:    var totalNumberOfColumns = 0;
057: 
058:    // Now, generate pairings for every flight
059:    for (var f=1; f <= nbFlights; f++) {
060:    
061:      var localGeneratePairingConfig =
062:       new IloOplRunConfiguration("GenerateOnePairing.mod");
063:      var localGeneratePairingModel = localGeneratePairingConfig.oplModel;
064:      generatePairingData.forcedCoveredFlight = f;
065:      localGeneratePairingModel.addDataSource(generatePairingData);
066:      localGeneratePairingModel.generate();     
067:      var localNumberOfPairings = 0;
068:      var localCP = localGeneratePairingConfig.cp;
069: 
070:      /// --- search loop for covering pairings
071:      /// seems better with nb agents ==1
072:      ///
073:          localCP.param.Workers = 1;
074:      localCP.startNewSearch();
075:      while (localCP.next()) {
076:        // Add the pairing to the list
077:        var currentPairingIndex = totalNumberOfColumns;
078:        /// 1. store the cost
079:        costs.add( currentPairingIndex, localGeneratePairingModel.costVar );
080:        /// 2. store the actual legs as triplets <#p, #l, flight>
081:        ///
082:        for (var l=1; l <= maxNbLegs; ++l) {   
083:          var flightRank  = localGeneratePairingModel.flightVars[l];
084:          var isActualLeg = localGeneratePairingModel.isActualLeg[l];
085:          //writeln("-- flight covered: ", f, ", #leg: ", l, ", f=", flightIndex);
086:          if ( isActualLeg < 0.1 ) break;  /// stop at dummy legs.
087:             
088:          /// BEWARE: variable is 1..F, but Opl.item() has C-convention
089:          legs.add(currentPairingIndex, l , flightRank);
090:        }//for           
091:        ++localNumberOfPairings;
092:        ++totalNumberOfColumns;
093:          
094:        if (localNumberOfPairings >= MAX_INITIAL_PAIRINGS_PER_FLIGHT) 
095:          break;
096:      }//while
097:      localCP.endSearch();
098:      if ( 0 == localNumberOfPairings) {
099:          writeln("** ERROR: cannot find any pairing for flight: ", f, " final problem will be infeasible");
100:          status = -2;
101:      }
102:      if ( localNumberOfPairings < MAX_INITIAL_PAIRINGS_PER_FLIGHT ) {
103:        ///  !!!
104:        /// This may be a problem: cannot find MAX_PAIRINGS pairings
105:        /// only a subset..        
106:        writeln("* warning: cannot find ", MAX_INITIAL_PAIRINGS_PER_FLIGHT, " pairings, only: ", localNumberOfPairings);
107:      }              
108:      if ( (f% PrintPeriod) == 0) {
109:        writeln("* initial generation: ", Opl.floor( (100 * f)/nbFlights +.5), "%, #columns: ", totalNumberOfColumns, ", time=", getElapsedTime(date0));
110:      }
111:    }
112:   
113:    writeln("* end initial pairings phase, time: " , time, "#columns=", totalNumberOfColumns, ", time=", getElapsedTime(date0), "s.");
114:    print_initial_pairings(legs, costs, "initial_pairings.dat");
115:    // this is the end
116:    
117:     writeln("*  script returns with code: ", status)
118:    status; 
119:  }
