012: 
013:  include "FlightPairing_data.mod";
014:  include "PairingGeneration_data.mod";
015:  
016:  using CP;
017:  
018:  int NbFlights  = card(flights);
019:  //int NbAirports = card(airports);
020:  int MaxLegs = parameters.maxNbLegs;
021:  int latestArrivalTime = max(f in flights) f.arrivalTime;
022:  int MaxWorkTime = parameters.workDurationMax;
023:  
024:  range FlightRange  = 1..NbFlights;
025:  range LegRange     = 1..MaxLegs;
026:  
027:  // time horizon is limited by latest arrival time.
028:  range horizon      = 0..latestArrivalTime;
029:  
030:  /// intermediate arrays for CP
031:  TFlightData FlightsByIndex[ f in FlightRange ] = item(flights,f-1);
032:  int arrivalTimes  [ f in FlightRange ] = FlightsByIndex[f].arrivalTime;
033:  int departureTimes[ f in FlightRange ] = FlightsByIndex[f].departureTime;
034:  int FlyingTimes[ f in FlightRange ] = FlightsByIndex[f].arrivalTime-FlightsByIndex[f].departureTime;
035:  
036:  int fromAirportIndex[ f in FlightRange] = ord(airports,<FlightsByIndex[f].fromAirportName>);
037:  int toAirportIndex  [ f in FlightRange] = ord(airports,<FlightsByIndex[f].toAirportName>);
038:  
039:  TAirport originAirport     [f in FlightRange] = item( airports, ord(airports,<FlightsByIndex[f].fromAirportName>) );
040:  
041:  int originTransitTimeMin[ f in FlightRange ] = originAirport[f].transitTimeMin;
042:  int originTransitTimeMax[ f in FlightRange ] = originAirport[f].transitTimeMax;
043:  
044:  /// external generation parameters
045: 
046:  int forcedCoveredFlight = ...;
047:  
048:  {TPairingLeg} legs;
049:  TPairingCost cost = <1, 0>;
050:  /**
051:   ** Pairings are modeled by a sequence of variables ranging over flights.
052:   ** 
053:   */
054:  dvar int flightVars[ LegRange ] in FlightRange;
055:  
056:  dvar boolean isActualLeg[ LegRange];
057:  
058:  dvar int+ startTimeVar in horizon;
059:  dvar int+ endTimeVar   in horizon;
060:  dvar int+ totalFlyingTime in 0..parameters.flyingDurationMax;
061:  
062:  /// pay
063:  dvar int+ costVar in parameters.minPay..999999999;
064:  dvar int+ workTimeCostVar;
065:  dvar int+ flyingTimeCostVar;
066:  
067:  dexpr int pairingLengthExp = sum(l in LegRange) isActualLeg[l];
068:  
069:  dvar int+ costVars[1..3];
070:  
071:  execute DEBUG {
072:    if ( PairingGenerationData.DebugMode >= 1 ) {
073:      writeln("-- searching for pairings with at most ", MaxLegs, " flights"); 
074:      if ( forcedCoveredFlight >= 1) {
075:        writeln("-- forcing coverage of flight #", forcedCoveredFlight, ", aka: ", Opl.item(flights, forcedCoveredFlight-1).name);   
076:      }
077:    }
078:  }
079:  
080: //minimize costVar;
081: subject to
082:  {
083:     // forbid one-leg pairings
084:     flightVars[1] != flightVars[2];
085:     isActualLeg[1] == 1;
086:     pairingLengthExp >= 3;
087:     
088:     // relate status vars to flight vars
089:     forall(l in LegRange: l > 1) {
090:       isActualLeg[l] == (flightVars[l] != flightVars[l-1]);    
091:     }
092:     
093:     // return home at the end of the chain
094:     fromAirportIndex[ flightVars[1] ] == toAirportIndex[ flightVars[ MaxLegs]];
095:     
096:     // either the chain ends or the airport chain is continuous
097:     forall(l in 2..MaxLegs) {
098:        (0 == isActualLeg[l])
099:        ||
100:        (fromAirportIndex[ flightVars[l] ] == toAirportIndex[ flightVars[l-1]]);
101:     }
102:   
103:     forall (l in 2..MaxLegs) {
104:        (0 == isActualLeg[l])    
105:      ||
106:      (
107:       departureTimes[ flightVars[l] ] >= arrivalTimes[ flightVars[l-1]]
108:       + originTransitTimeMin[ flightVars[l] ]
109:       );
110:      }
111:     forall (l in 2..MaxLegs) {       
112:         (isActualLeg[l]==0)
113:       || 
114:       (
115:       departureTimes[ flightVars[l] ] <= arrivalTimes[ flightVars[l-1]]
116:       + originTransitTimeMax[ flightVars[l] ]
117:       );
118:     }
119:     /// force coverage of (some) flights.
120:     if  (forcedCoveredFlight >= 1) {
121:       count(flightVars,forcedCoveredFlight) >= 1;   
122:     }
123:     
124:     /// propagate end of chain to the right
125:     forall (l in LegRange: 1 < l) {
126:        isActualLeg[l-1] >= isActualLeg[l];
127:     }
128:     
129:     // total flying time is counted only on "actual" legs
130:     totalFlyingTime == 
131:     FlyingTimes[ flightVars[1]]
132:     + sum(l in 2..MaxLegs) ( FlyingTimes[ flightVars[l] ] * isActualLeg[l]);
133:     
134:     startTimeVar == departureTimes[ flightVars[1]];
135:     endTimeVar   == arrivalTimes  [ flightVars[MaxLegs] ];
136:     
137:     (endTimeVar <= startTimeVar + MaxWorkTime);
138:     
139:     flyingTimeCostVar >= parameters.flyingPayRate * totalFlyingTime;
140:     flyingTimeCostVar <= parameters.flyingPayRate * totalFlyingTime+1;
141:     workTimeCostVar >= parameters.workPayRate * (endTimeVar - startTimeVar);
142:     workTimeCostVar <= parameters.workPayRate * (endTimeVar - startTimeVar)+1;
143:     
144:     /// all that pain to use max[]
145:     costVars[1] == parameters.minPay;
146:     costVars[2] == workTimeCostVar;
147:     costVars[3] == flyingTimeCostVar;
148:     
149:     costVar == max(j in 1..3) costVars[j];
150:  }
151:  
152:  execute DISPLAY_SOLUTION {
153:    var lastTime = -1;
154:    var lastFlight = null;
155:    
156:    for (var l in LegRange) {
157:      var flightRank = flightVars[l];
158:      var flight = FlightsByIndex[flightRank];
159:      legs.add(1, l, flightRank);
160:      
161:      if ( flight == lastFlight ) break;
162:      writeln("* leg #", l, ", f#", flightRank, flight.name, ", from ", flight.fromAirportName, " to ", flight.toAirportName);
163:      var idle = lastTime>=0? flight.departureTime - lastTime: 0;
164:      //writeln("* idle: ", idle, ", fly time=", flight.arrivalTime-flight.departureTime);
165:      lastTime = flight.arrivalTime;
166:      lastFlight = flight;
167:    }
168:    cost.cost = costVar;
169:    writeln(" work=", (endTimeVar - startTimeVar), ", flying=", totalFlyingTime);
170:    writeln(" cost=", costVar, ", work cost=", workTimeCostVar, ", fly cost=", flyingTimeCostVar);
171:  }
