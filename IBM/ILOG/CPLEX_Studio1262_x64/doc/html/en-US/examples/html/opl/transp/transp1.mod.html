12: {string} Cities =...;
13: {string} Products = ...;
14: float Capacity = ...;
15: 
16: float Supply[Products][Cities] = ...;
17: float Demand[Products][Cities] = ...;
18: assert
19:   forall(p in Products)
20:     sum(o in Cities) Supply[p][o] == sum(d in Cities) Demand[p][d];
21: float Cost[Products][Cities][Cities] = ...;
22: 
23: dvar float+ Trans[Products][Cities][Cities];
24: 
25: 
26: minimize
27:   sum( p in Products , o , d in Cities ) 
28:     Cost[p][o][d] * Trans[p][o][d];
29:    
30: subject to {
31:   forall( p in Products , o in Cities )
32:     ctSupply:  
33:       sum( d in Cities ) 
34:         Trans[p][o][d] == Supply[p][o];
35:   forall( p in Products , d in Cities ) 
36:     ctDemand:
37:       sum( o in Cities ) 
38:         Trans[p][o][d] == Demand[p][d];
39:    forall( o , d in Cities )
40:      ctCapacity:
41:        sum( p in Products ) 
42:          Trans[p][o][d] <= Capacity;
43: }  
44: 
45: execute DISPLAY {
46:   writeln("trans = ",Trans);
47: }

52:         string City1;
53:         string City2;
54:         float Trans;    
55: };
