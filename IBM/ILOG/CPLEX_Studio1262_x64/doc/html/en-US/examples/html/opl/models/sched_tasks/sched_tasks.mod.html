012: // Problem Description : workforce scheduling problem.
013: //
014: // A set of resources (tuple ResourceDat) of different types are available to perform some requests.
015: // Different types are requests are considered (tuple RequestType). 
016: // A given type of request can be decomposed into a set of tasks (tuple Recipe) and 
017: // some temporal dependencies between those tasks (tuple Dependency).
018: // Each task is associated a processing time (tuple Task) and a set of resource requirements (tuple Requirement).
019: // A requirement consists of a task type, a resource type and a quantity 
020: // (number of resources of the specified type to be used for executing this type of task).
021: //
022: // The objective is to schedule a set of requests with individual due dates so as to minimize the total tardiness. 
023: //
024: // Model and Constraints
025: //
026: // Each request, task and possible allocation of a task to a resource for a requirement 
027: // is modeled as an interval variable. 
028: // A request spans its tasks. 
029: // Allocations are optional.
030: // Each requirement posts a generalized alternative constraints between 
031: // the task and the set of possible allocations for this requirement. 
032: // The cardinality of this generalized alternative is the number of required resources. 
033: //
034: // Each resource is a sequence of its non-overlaping allocations.
035: //
036: // The objective is to minimize total tardiness.
037: //
038: // Redundant cumul.
039: //
040: // In this model, a redundant cumul function is used to globally constrain 
041: // the number of resource of a certain type simultaneously used by the tasks. 
042: // This cumul is limited by the number of resources of the resource type.
043: // These redundant cumuls may help in some models as they enforce a stronger inference 
044: // in the engine while the whole set of resources for the tasks is still not completely chosen. 
045: // For more complex problems, e.g resources with several resource types / skills, 
046: // other partitions of the resource set may define efficient redundant cumul. 
047: 
048: using CP;
049: 
050: // Data for resources, requests and tasks
051: 
052: tuple ResourceDat {
053:   key int id;
054:   string  type;
055:   string name;
056: };
057: 
058: tuple RequestDat {
059:   key int id;
060:   string  type;
061:   int     duedate;
062:   string  name;
063: };
064: 
065: tuple TaskDat {
066:   key int id;
067:   string  type;
068:   int     ptime;
069:   string  name;
070: };
071: 
072: {RequestDat}  requests  = ...;
073: {ResourceDat} resources = ...;
074: {TaskDat}     tasks     = ...;
075: 
076: {string} resourceTypes = { r.type | r in resources };
077: 
078: // Data for template recipes, dependencies and requirements
079: 
080: tuple Recipe {
081:   string request;
082:   string task;
083: };
084: 
085: tuple Dependency {
086:   string request;
087:   string taskb;
088:   string taska;
089:   int    delay;
090: };
091: 
092: tuple Requirement {
093:   string task;
094:   string resource;
095:   int    quantity;
096: };
097: 
098: {Recipe}      recipes      = ...;
099: {Dependency}  dependencies = ...;
100: {Requirement} requirements = ...;
101: 
102: // Set of operations (tasks of a request) and allocations (operation on a possible resource)
103: 
104: tuple Operation {
105:   RequestDat request;
106:   TaskDat    task;
107: };
108: 
109: tuple Allocation {
110:   Operation   dmd;
111:   Requirement req;
112:   ResourceDat resource;
113: };
114: 
115: {Operation} operations = 
116:   { <r, t> | r in requests,  m in recipes, t in tasks : 
117:    r.type == m.request && t.type == m.task};
118: 
119: {Allocation} allocs = 
120:   { <o, m, r> | o in operations, m in requirements, r in resources : 
121:    o.task.type == m.task && r.type == m.resource};
122: 
123: dvar interval tirequests[requests];
124: dvar interval tiops[o in operations] size o.task.ptime;
125: dvar interval tiallocs[allocs] optional;
126: 
127: dvar sequence workers[r in resources] in all(a in allocs: a.resource == r) tiallocs[a];
128: 
129: int levels[rt in resourceTypes] = sum(r in resources : r.type == rt) 1;
130: 
131: cumulFunction cumuls[rt in resourceTypes] =
132:   sum(rc in requirements, o in operations : rc.resource == rt && o.task.type == rc.task) pulse(tiops[o], rc.quantity);
133: 
134: minimize sum(t in requests) maxl(0, endOf(tirequests[t]) - t.duedate);
135: subject to {
136:   forall(r in requests) {
137:     span(tirequests[r], all(o in operations : o.request == r) tiops[o]);
138:     forall (o in operations : o.request == r) {
139:       forall (rc in requirements : rc.task == o.task.type) {
140:         alternative(tiops[o], all(a in allocs : a.req == rc && a.dmd == o) tiallocs[a], rc.quantity);
141:       }        
142:       forall(tc in dependencies: tc.request == r.type && tc.taskb == o.task.type) {
143:         forall(o2 in operations : o2.request == r && tc.taska == o2.task.type) {
144:           endBeforeStart(tiops[o], tiops[o2], tc.delay);    
145:         }
146:       }
147:     }   
148:   }
149:   forall(r in resources) {
150:     noOverlap(workers[r]);
151:   }    
152:   forall(r in resourceTypes: levels[r] > 1) {
153:     cumuls[r] <= levels[r];
154:   }    
155: };
