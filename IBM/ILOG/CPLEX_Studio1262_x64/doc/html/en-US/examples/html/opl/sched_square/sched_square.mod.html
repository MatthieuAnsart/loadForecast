12: using CP;
13: 
14: int SizeSquare = 112;
15: int NbSquares  = 21;
16: 
17: execute{
18:         }
19: 
20: range Squares = 1..NbSquares;
21: 
22: int Size[Squares] = [50,42,37,35,33,29,27,25,24,19,18,17,16,15,11,9,8,7,6,4,2];
23: 
24: dvar interval x[s in Squares] in 0..SizeSquare size Size[s];
25: dvar interval y[s in Squares] in 0..SizeSquare size Size[s];
26: 
27: cumulFunction rx = sum(s in Squares) pulse(x[s], Size[s]);
28: cumulFunction ry = sum(s in Squares) pulse(y[s], Size[s]);
29: 
30: execute {
31:   var f = cp.factory;
32:   cp.setSearchPhases(f.searchPhase(x),   
33:                      f.searchPhase(y));
34: };
35: 
36: constraints {
37:    alwaysIn(rx,0,SizeSquare,SizeSquare,SizeSquare);
38:    alwaysIn(ry,0,SizeSquare,SizeSquare,SizeSquare);
39:    forall(ordered i, j in Squares)
40:      endOf(x[i]) <= startOf(x[j]) ||
41:      endOf(x[j]) <= startOf(x[i]) ||
42:      endOf(y[i]) <= startOf(y[j]) ||
43:      endOf(y[j]) <= startOf(y[i]);
44: };
