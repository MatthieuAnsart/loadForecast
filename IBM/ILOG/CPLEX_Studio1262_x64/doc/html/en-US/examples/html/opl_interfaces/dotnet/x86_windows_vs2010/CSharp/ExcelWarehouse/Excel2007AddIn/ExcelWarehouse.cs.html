120:         private void SolveProblem(Excel.Worksheet sheet, System.IO.StringWriter errHandlerErrors)
121:         {
122:             // actually solve the problem with OPL
123:             OplFactory.DebugMode = true;
124:             OplFactory oplF = new OplFactory();
125:             OplErrorHandler errHandler = oplF.CreateOplErrorHandler(errHandlerErrors);
126:             OplModelSource modelSource = oplF.CreateOplModelSourceFromString(GetModelText(), "warehouse");
127:             OplSettings settings = oplF.CreateOplSettings(errHandler);
128:             OplModelDefinition def = oplF.CreateOplModelDefinition(modelSource, settings);
129:             Cplex cplex = oplF.CreateCplex();
130:             OplModel opl = oplF.CreateOplModel(def, cplex);
131:             OplDataSource dataSource = new MyParams(oplF, sheet);
132:             opl.AddDataSource(dataSource);
133:             opl.Generate();
134:             if (cplex.Solve())
135:             {
136:                 setValue(sheet, "B10", "" + opl.Cplex.ObjValue);
137:             }
138:             else
139:             {
140:                 setValue(sheet, "B10", "No solution");
141:             }
142:             oplF.End();
143:         }

191:         /**
192:          * This is the warehouse OPL model:
193:          */
194:         static String GetModelText()
195:         {
196:             String model = "";
197:             model += "int   fixed        = ...;";
198:             model += "int   nbWarehouses = ...;";
199:             model += "int   nbStores     = ...;";
200:             model += "int   disaggregate = ...;";
201:             model += "assert nbStores > nbWarehouses;";
202: 
203:             model += "range Warehouses = 1..nbWarehouses;";
204:             model += "range Stores     = 1..nbStores;";
205: 
206:             model += "int capacity[w in Warehouses] = nbStores div nbWarehouses + w mod (nbStores div nbWarehouses);";
207:             model += "int supplyCost[s in Stores][w in Warehouses] = 1+((s+10*w) mod 100);";
208: 
209:             model += "dvar float open[Warehouses] in 0..1;";
210:             model += "dvar float supply[Stores][Warehouses] in 0..1;";
211: 
212:             model += "minimize ";
213:             model += "sum(w in Warehouses) fixed * open[w] +";
214:             model += "sum(w in Warehouses, s in Stores) supplyCost[s][w] * supply[s][w];";
215: 
216:             model += "constraints {";
217:             model += "  forall(s in Stores)";
218:             model += "    sum(w in Warehouses) supply[s][w] == 1;";
219:             model += "  forall(w in Warehouses)";
220:             model += "    sum(s in Stores) supply[s][w] <= open[w]*capacity[w];";
221:             model += "  if (disaggregate == 1) {";
222:             model += "   forall(w in Warehouses, s in Stores)";
223:             model += "      supply[s][w] <= open[w];";
224:             model += "  }";
225:             model += "}";
226:             return model;
227:         }
