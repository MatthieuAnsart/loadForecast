19: int NumNodes = ...;   // Number of nodes
20: range Nodes = 1..NumNodes;
21: 
22: // Get the supply (positive) and demand (negative)
23: // at each node
24: 
25: int SupDem[Nodes] = ...;
26: 
27: // Create a record to hold information about each arc
28: tuple arc {
29:    key int fromnode;
30:    key int tonode;
31:    float cost;
32:    float ub;
33: }
34: 
35: // Get the set of arcs
36: 
37: {arc} Arcs = ...;
38: 
39: // The network flow model has decision variables indexed on 
40: // the arcs.
41: 
42: dvar float+ Flow[a in Arcs] in 0 .. a.ub;
43: 
44: dexpr float TotalFlow = sum (a in Arcs) a.cost * Flow[a];
45: minimize TotalFlow;
46: subject to {
47:    // Preserve flows at each node.  Note the use of slicing
48:    forall (i in Nodes)
49:      ctNodeFlow:
50:       sum (<i,j,c,ub> in Arcs) Flow[<i,j,c,ub>]
51:     - sum (<j,i,c,ub> in Arcs) Flow[<j,i,c,ub>] == SupDem[i];
52: }
53: 
54: tuple FlowSolutionT{ 
55:         arc Arcs;       
56:         float value; 
57: };
58: {FlowSolutionT} FlowSolution = {<i0,Flow[i0]> | i0 in Arcs};
59: 
60: 
61: execute DISPLAY {
62:    writeln("\n<from node,to node,Flow[a]>\n");
63:    for(var a in Arcs)
64:       if(Flow[a] > 0)
65:          writeln("<",a.fromnode,",",a.tonode,",",Flow[a],">");
66: }
