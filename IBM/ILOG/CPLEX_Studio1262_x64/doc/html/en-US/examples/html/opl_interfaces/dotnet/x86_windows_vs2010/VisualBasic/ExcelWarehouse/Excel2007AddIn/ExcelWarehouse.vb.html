092:     Private Sub SolveProblem(ByVal sheet As Excel.Worksheet, ByVal errHandlerErrors As StringWriter)
093:         ' actually solve the problem with OPL
094:         OplFactory.DebugMode = True
095:         Dim oplF As OplFactory = New OplFactory()
096:         Dim errHandler As OplErrorHandler = oplF.CreateOplErrorHandler(errHandlerErrors)
097:         Dim modelSource As OplModelSource = oplF.CreateOplModelSourceFromString(GetModelText(), "warehouse")
098:         Dim settings As OplSettings = oplF.CreateOplSettings(errHandler)
099:         Dim def As OplModelDefinition = oplF.CreateOplModelDefinition(modelSource, settings)
100:         Dim cplex As Cplex = oplF.CreateCplex()
101:         Dim opl As OplModel = oplF.CreateOplModel(def, cplex)
102:         Dim dataSource As OplDataSource = New MyParams(oplF, sheet)
103:         opl.AddDataSource(dataSource)
104:         opl.Generate()
105:         If (cplex.Solve()) Then
106:             setValue(sheet, "B10", "" + opl.Cplex.ObjValue.ToString)
107:         Else
108:             setValue(sheet, "B10", "No solution")
109:         End If
110:         oplF.End()
111:     End Sub

113:     Function GetModelText() As String
114:         Dim model As String
115:         model = "int   fixed        = ...;" & vbCrLf & _
116:                 "int   nbWarehouses = ...;" & vbCrLf & _
117:                 "int   nbStores     = ...;" & vbCrLf & _
118:                 "int   disaggregate = ...;" & vbCrLf & _
119:                 "assert nbStores > nbWarehouses;" & vbCrLf & _
120:                 "range Warehouses = 1..nbWarehouses;" & vbCrLf & _
121:                 "range Stores     = 1..nbStores;" & vbCrLf & _
122:                 "int capacity[w in Warehouses] = nbStores div nbWarehouses + w mod (nbStores div nbWarehouses);" & vbCrLf & _
123:                 "int supplyCost[s in Stores][w in Warehouses] = 1+((s+10*w) mod 100);" & vbCrLf & _
124:                 "dvar float open[Warehouses] in 0..1;" & vbCrLf & _
125:                 "dvar float supply[Stores][Warehouses] in 0..1;" & vbCrLf & _
126:                 "minimize " & vbCrLf & _
127:                 "sum(w in Warehouses) fixed * open[w] +" & vbCrLf & _
128:                 "sum(w in Warehouses, s in Stores) supplyCost[s][w] * supply[s][w];" & vbCrLf & _
129:                 "constraints {" & vbCrLf & _
130:                 "  forall(s in Stores)" & vbCrLf & _
131:                 "    sum(w in Warehouses) supply[s][w] == 1;" & vbCrLf & _
132:                 "  forall(w in Warehouses)" & vbCrLf & _
133:                 "    sum(s in Stores) supply[s][w] <= open[w]*capacity[w];" & vbCrLf & _
134:                 "  if (disaggregate == 1) {" & vbCrLf & _
135:                 "   forall(w in Warehouses, s in Stores)" & vbCrLf & _
136:                 "      supply[s][w] <= open[w];" & vbCrLf & _
137:                 "  }" & vbCrLf & _
138:                 "}"
139:         Return model
140:     End Function
