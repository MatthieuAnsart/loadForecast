012: {string} Products = ...;
013: 
014: int NbMonths   = ...;
015: range Months = 1..NbMonths;
016: float Cost[Months][Products] = ...;
017: 
018: dvar float+ Produce[Months];
019: dvar float+ Use[Months][Products];
020: dvar float+ Buy[Months][Products];
021: dvar float Store[Months][Products] in 0..1000;
022: 
023: 
024: maximize 
025:   sum( m in Months ) 
026:     (150 * Produce[m] 
027:     - sum( p in Products ) 
028:       Cost[m][p] * Buy[m][p] 
029:     - 5 * sum( p in Products ) 
030:       Store[m][p]);
031: subject to {
032:   forall( p in Products )
033:     ctStore:
034:       Store[NbMonths][p] == 500;
035:   forall( m in Months ) {
036:     ctUse1:
037:       Use[m]["v1"] + Use[m]["v2"] <= 200;
038:     ctUse2:                
039:       Use[m]["o1"] + Use[m]["o2"] + Use[m]["o3"] <= 250;
040:     ctUse3:
041:       3 * Produce[m] <=
042:             8.8 * Use[m]["v1"] + 6.1 * Use[m]["v2"] +
043:             2   * Use[m]["o1"] + 4.2 * Use[m]["o2"] + 5 * Use[m]["o3"];
044:      ctUse4:
045:        8.8 * Use[m]["v1"] + 6.1 * Use[m]["v2"] +
046:             2   * Use[m]["o1"] + 4.2 * Use[m]["o2"] + 5 * Use[m]["o3"]
047:             <= 6 * Produce[m];
048:             
049:      ctUse5:
050:        Produce[m] == sum( p in Products ) Use[m][p];
051:    }
052:   forall( m in Months )
053:     forall( p in Products ) {
054:       ctUse6:  
055:         if (m == 1) {
056:           500 + Buy[m][p] == Use[m][p] + Store[m][p];
057:         }
058:         else {
059:           Store[m-1][p] + Buy[m][p] == Use[m][p] + Store[m][p];
060:         }
061:     }
062:     forall( m in Months ) {
063:       // Using some constraints as boolean expressions to state that at least
064:       // 2 of the given 5 constraints must be true.
065:       ctUse7:  
066:         (Use[m]["v1"] == 0) + (Use[m]["v2"] == 0) + (Use[m]["o1"] == 0) +
067:                    (Use[m]["o2"] == 0) + (Use[m]["o3"] == 0) >= 2;
068: 
069:       // Using the "or" operator, set each Use variable to be
070:       // zero or greather than 20.
071:       forall( p in Products )
072:         ctUse8:    
073:           (Use[m][p] == 0) || (Use[m][p] >= 20);
074: 
075:       // Using "or" and "implication" operator, set that if one of 2 given products 
076:       // is used more than 20 then a third one must be used more than 20 too.
077:       ctUse9:
078:         (Use[m]["v1"] >= 20) || (Use[m]["v2"] >= 20) => Use[m]["o3"] >= 20;
079:   }
080: };
081: 
082: // Expected result : 100278.70
083:

086:         int Months; 
087:         float value; 
088: };
089: {ProduceSolutionT} ProduceSolution = {<i0,Produce[i0]> | i0 in Months};
090: tuple BuySolutionT{ 
091:         int Months;

093:         float value; 
094: };
095: {BuySolutionT} BuySolution = {<i0,i1,Buy[i0][i1]> | i0 in Months,i1 in Products};

097:         int Months; 
098:         string Products; 
099:         float value; 
100: };

102: tuple UseSolutionT{ 
103:         int Months; 
104:         string Products; 
105:         float value;

107: {UseSolutionT} UseSolution = {<i0,i1,Use[i0][i1]> | i0 in Months,i1 in Products};
