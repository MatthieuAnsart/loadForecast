12: /* MIQP: Mixed Integer Quadratic Constraint
13:  *
14:  * Example from CPLEX 9.0: iloqmiqpex1.cpp 
15:  */
16: dvar float x[0..2] in 0..40;
17: dvar int x3 in 0..3;
18: 
19: maximize
20:   x[0] + 2 * x[1] + 3 * x[2] + x3
21:        - 0.5 * ( 33*x[0]^2 + 22*x[1]^2 + 11*x[2]^2 
22:                - 12*x[0]*x[1] -23*x[1]*x[2] );
23: 
24: subject to {
25:   ct1:
26:     - x[0] +     x[1] + x[2] + 10 * x3 <= 20;
27:   ct2:
28:       x[0] - 3 * x[1] + x[2]           <= 30;
29:   ct3:
30:                  x[1]        - 3.5* x3 == 0;
31: }
