012: /******************************************************************************
013:  * 
014:  * OPL Model for Telecommunication Route Generation
015:  * 
016:  * Construct a "route" from a set of end-to-end paths so that the collection
017:  * of paths satisfies the blocking criterion at a minimum cost.  This extends
018:  * the two-arc path optimization algorithm in chapter 4 of "Dynamic Routing in 
019:  * Telecommunications Networks" by Gerald R. Ash (McGraw Hill).
020:  * 
021:  * This is a dynamic programming model with two global constraints:
022:  *      1. All paths are different
023:  *      2. Blocking criterion
024:  * The "allDifferent" constraint is implemented explicitly to allow duplicates
025:  * of a dummy value at the end of the sequence.
026:  * 
027:  * Pseudorandom cost and blocking data are generated within the model.
028:  * 
029: ******************************************************************************/
030: 
031: using CP;
032: 
033: execute
034: {
035: // This makes sure we'll get the exact minimum
036:                 cp.param.optimalityTolerance = 1.0;
037:         cp.param.relativeOptimalityTolerance = 0.0; 
038: }    
039: 
040: /******************************************************************************
041:  * DATA DECLARATIONS
042:  ******************************************************************************/
043: 
044: int nNodes          = 500;          // Number of nodes
045: int SCALE           = 1000;         // Integer scaling factor
046: int blkTol          = -6;           // Blocking criterion
047: 
048: range Nodes         = 1..nNodes;
049: range Paths         = 0..nNodes;    // Only consider 2-arc paths
050: 
051: // Blocking percentages
052: int blk[p in Paths] = (p==0)?0:(1+rand(10));   
053: // Logs of blocking percentages
054: int lbk[p in Paths] = ftoi(round(log(0.01*((p==0)?(1+rand(10)):blk[p]))*SCALE));
055: // Costs 
056: int cst[p in Paths] = (p==0)?0:(10+rand(90));                
057: 
058: /*
059:  * Initialize blocking percentages and path costs
060:  * with pseudorandom data.
061:  */
062: 
063: 
064: // Summary data
065: int minRoute    = ftoi(ceil(1.0*SCALE*blkTol/(min (i in Paths) lbk[i])));
066: int maxCost     = max (i in Nodes) cst[i];
067: int rteSize     = 2+minRoute;
068: range Route     = 1..rteSize;
069: 
070: 
071: /******************************************************************************
072:  * DECISION VARIABLES
073:  ******************************************************************************/
074: range R = 1..rteSize+1;
075: dvar int     rte[Route] in Paths;
076: dvar int     cost[R]  in 0..SCALE*maxCost;
077: 
078: 
079: /******************************************************************************
080:  * MODEL
081:  ******************************************************************************/
082: 
083: minimize cost[1];
084: subject to {
085:     
086:     // Recursive objective function
087:     cost[rteSize+1] == 0;
088:     forall (i in Route)
089:       cost[i] == ( SCALE*(100-blk[rte[i]])*cst[rte[i]]
090:                    + blk[rte[i]]*cost[i+1] ) div 100;
091:               
092:     // Path constraints: All paths are different except
093:     // dummy paths at the end.  This comes from simplifying
094:     // the two constraints:
095:    
096:     forall (ordered i,j in Route)
097:         (rte[i] != 0 && rte[i] != rte[j]) || (rte[j] == 0);
098:     
099:     // Blocking criterion: Joint probability that all paths are
100:     // blocked is at most 10^blkTol.  Take the log of both sides
101:     // to avoid underflow.
102:     sum (i in Route) lbk[rte[i]] <= SCALE*blkTol;
103:               
104:     // Redundant constraint: 
105:     // Eliminate dummy paths from the start of the route
106:     forall (i in 1..minRoute) rte[i] != 0;
107: 
108:     // Redundant constraint:
109:     // The expected cost is at least the minimum cost of a
110:     // single path
111:     cost[1] >= SCALE * (min (i in Nodes) cst[i]);
112: };
113: 
114: 
115: tuple costSolutionT{ 
116:         int R; 
117:         int value; 
118: };
119: {costSolutionT} costSolution = {<i0,cost[i0]> | i0 in R};
120: tuple rteSolutionT{ 
121:         int Route; 
122:         int value; 
123: };
124: {rteSolutionT} rteSolution = {<i0,rte[i0]> | i0 in Route};
125: 
126: execute{ 
127:         writeln(costSolution);
128:         writeln(rteSolution);
129: }
