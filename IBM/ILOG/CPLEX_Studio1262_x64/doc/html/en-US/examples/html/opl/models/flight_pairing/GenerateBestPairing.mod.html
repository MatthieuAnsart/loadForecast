011: 
012:  include "FlightPairing_data.mod";
013:  include "PairingGeneration_data.mod";
014:  include "FlightCoverDual_data.mod";
015:  
016:  using CP;
017:  
018:  int NbFlights  = card(flights);
019:  //int NbAirports = card(airports);
020:  int MaxLegs = parameters.maxNbLegs;
021:  int latestArrivalTime = max(f in flights) f.arrivalTime;
022:  int MaxWorkTime = parameters.workDurationMax;
023:  
024:  range FlightRange  = 1..NbFlights;
025:  range LegRange     = 1..MaxLegs;
026:  
027:  // time horizon is limited by latest arrival time.
028:  range horizon      = 0..latestArrivalTime;
029:  
030:  /// intermediate arrays for CP
031:  TFlightData FlightsByIndex[ f in FlightRange ] = item(flights,f-1);
032:  int arrivalTimes  [ f in FlightRange ] = FlightsByIndex[f].arrivalTime;
033:  int departureTimes[ f in FlightRange ] = FlightsByIndex[f].departureTime;
034:  int FlyingTimes[ f in FlightRange ] = FlightsByIndex[f].arrivalTime-FlightsByIndex[f].departureTime;
035:  
036:  int fromAirportIndex[ f in FlightRange] = ord(airports,<FlightsByIndex[f].fromAirportName>);
037:  int toAirportIndex  [ f in FlightRange] = ord(airports,<FlightsByIndex[f].toAirportName>);
038:  
039:  TAirport originAirport     [f in FlightRange] = item( airports, ord(airports,<FlightsByIndex[f].fromAirportName>) );
040:  //TAirport destinationAirport[f in FlightRange] = item( airports, ord(airports,<FlightsByIndex[f].originAirportName>) );
041:   
042:  int originTransitTimeMin[ f in FlightRange ] = originAirport[f].transitTimeMin;
043:  int originTransitTimeMax[ f in FlightRange ] = originAirport[f].transitTimeMax;
044:  
045: {TPairingLeg} genLegs;
046: TPairingCost genCost = <1, 0>;
047:  
048:  /**
049:   ** Pairings are modeled by a sequence of variables ranging over flights.
050:   ** 
051:   */
052:  dvar int flightVars[ LegRange ] in FlightRange;
053:  
054:  dvar boolean isActualLeg[ LegRange];
055:  
056:  dvar int+ startTimeVar in horizon;
057:  dvar int+ endTimeVar   in horizon;
058:  dvar int+ totalFlyingTime in 0..parameters.flyingDurationMax;
059:  
060:  /// pay
061:  dvar int+ costVar in parameters.minPay..999999999;
062:  dvar int+ workTimeCostVar;
063:  dvar int+ flyingTimeCostVar;
064:  
065:  dvar int+ costVars[1..3];
066:  
067: /// input from master LP model.
068: {TFlightCoverDual} coverct_duals = ...;
069: 
070: float duals[ FlightRange ] = [ fcd.flightRank : fcd.coverCtDual | fcd in coverct_duals ];
071: 
072: dexpr float dualObj =
073:   sum (f in FlightRange) duals[f] * (count(flightVars,f)>=1);
074: 
075: execute LIMIT_CP {
076:   cp.param.branchLimit = thisOplModel.PairingGenerationData.SlaveModelBranchLimit;
077:   cp.param.Workers = 1;
078: }
079: 
080: minimize 1 - dualObj;
081: 
082: subject to
083:  {
084:     // forbid one-leg pairings
085:     flightVars[1] != flightVars[2];
086:     isActualLeg[1] == 1;
087:     
088:     // sum(l in LegRange) isActualLeg[l] >= 7;
089:     
090:     // relate status vars to flight vars
091:     forall(l in LegRange: l > 1) {
092:       isActualLeg[l] == (flightVars[l] != flightVars[l-1]);    
093:     }
094:     
095:     // return home at the end of the chain
096:     fromAirportIndex[ flightVars[1] ] == toAirportIndex[ flightVars[ MaxLegs]];
097:     
098:     // either the chain ends or the airport chain is continuous
099:     forall(l in 2..MaxLegs) {
100:        (0 == isActualLeg[l])
101:        ||
102:        (fromAirportIndex[ flightVars[l] ] == toAirportIndex[ flightVars[l-1]]);
103:     }
104:   
105:     forall (l in 2..MaxLegs) {
106:        (0 == isActualLeg[l])    
107:      ||
108:      (
109:       departureTimes[ flightVars[l] ] >= arrivalTimes[ flightVars[l-1]]
110:       + originTransitTimeMin[ flightVars[l] ]
111:       );
112:      }
113:     forall (l in 2..MaxLegs) {       
114:         (isActualLeg[l]==0)
115:       || 
116:       (
117:       departureTimes[ flightVars[l] ] <= arrivalTimes[ flightVars[l-1]]
118:       + originTransitTimeMax[ flightVars[l] ]
119:       );
120:     }
121:     
122:     /// propagate end of chain to the right
123:     forall (l in LegRange: 1 < l) {
124:        isActualLeg[l-1] >= isActualLeg[l];
125:     }
126:     
127:     // total flying time is counted only on "actual" legs
128:     totalFlyingTime == 
129:     FlyingTimes[ flightVars[1]]
130:     + sum(l in 2..MaxLegs) ( FlyingTimes[ flightVars[l] ] * isActualLeg[l]);
131:     
132:     startTimeVar == departureTimes[ flightVars[1]];
133:     endTimeVar   == arrivalTimes  [ flightVars[MaxLegs] ];
134:     
135:     (endTimeVar <= startTimeVar + MaxWorkTime);
136:     
137:     flyingTimeCostVar >= parameters.flyingPayRate * totalFlyingTime;
138:     flyingTimeCostVar <= parameters.flyingPayRate * totalFlyingTime+1;
139:     workTimeCostVar >= parameters.workPayRate * (endTimeVar - startTimeVar);
140:     workTimeCostVar <= parameters.workPayRate * (endTimeVar - startTimeVar)+1;
141:     
142:     /// all that pain to use max[]
143:     costVars[1] == parameters.minPay;
144:     costVars[2] == workTimeCostVar;
145:     costVars[3] == flyingTimeCostVar;
146:     
147:     costVar == max(j in 1..3) costVars[j];
148:  }
149:  
150:  
151:  
152:  execute DISPLAY_SOLUTION {
153:    var lastTime = -1;
154:    var lastFlight = null;
155:    genCost.cost = costVar;
156:    
157:    for (var l in LegRange) {
158:      var flightRank = flightVars[l];
159:      var flight = FlightsByIndex[flightRank];
160:      genLegs.add(1, l, flightRank);
161:      
162:      if ( flight == lastFlight ) break;
163:      //writeln("* leg #", l, "f: ", flightRank, ", flight: ", flight.name, ", from ", flight.fromAirportName, " to ", flight.toAirportName);
164:      var idle = lastTime>=0? flight.departureTime - lastTime: 0;
165:      //writeln("* idle: ", idle, ", fly time=", flight.arrivalTime-flight.departureTime);
166:      lastTime = flight.arrivalTime;
167:      lastFlight = flight;
168:    } 
169:    //writeln(" work=", (endTimeVar - startTimeVar), ", flying=", totalFlyingTime);
170:    //writeln(" cost=", costVar, ", dual:", dualObj);
171:  }
