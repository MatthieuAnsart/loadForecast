020: int NbLands = ...;
021: range Lands = 1..NbLands;
022: 
023: int NbYears = ...;
024: range Years = 1..NbYears;
025: 
026: int NbAges = ...;
027: range Ages = 1..NbAges;
028: // Heifers are cows up to age 2
029: // Productive cows are past heifer age but not yet at selling age
030: range CowAges = 2..NbAges-1;
031: 
032: float InitCows[Ages] = ...;
033: float HeifSurvival = ...;
034: float CowSurvival = ...;
035: float CalfRate = ...;
036: float HeifFraction = ...;
037: float InitCap = ...;
038: float GrPerCow = ...;
039: float SbPerCow = ...;
040: float GrPerAcre[Lands] = ...;
041: float GrAcre[Lands] = ...;
042: float SbPerAcre = ...;
043: float HeifAcre = ...;
044: float CowAcre = ...;
045: float Acres = ...;
046: float HeifLabor = ...;
047: float CowLabor = ...;
048: float Labor = ...;
049: float GrLabor = ...;
050: float SbLabor = ...;
051: float DownCowChg = ...;
052: float UpCowChg = ...;
053: float BullockSellPrice = ...;
054: float HeifSellPrice = ...;
055: float CowSellPrice = ...;
056: float MilkPerCowPrice = ...;
057: float GrSellPrice = ...;
058: float SbSellPrice = ...;
059: float GrBuyPrice = ...;
060: float SbBuyPrice = ...;
061: float LbrFixCost = ...;
062: float LbrCost = ...;
063: float HeifCost = ...;
064: float CowCost = ...;
065: float GrGrowCost = ...;
066: float SbGrowCost = ...;
067: int   LoanTerm = ...;
068: float Repay = ...;
069: 
070: float InitCowsTotal = sum(j in CowAges) InitCows[j];
071: 
072: dvar float+ GrGrow[l in Lands][y in Years] in 0..GrPerAcre[l]*GrAcre[l];
073: dvar float+ GrBuy[Years];
074: dvar float+ GrSell[Years];
075: dvar float+ SbGrow[Years];
076: dvar float+ SbBuy[Years];
077: dvar float+ SbSell[Years];
078: dvar float+ Lbr[Years];
079: dvar float+ CapAdd[Years];
080: dvar float+ HeifSell[Years];
081: dvar float+ Cow[Ages,Years];
082: dvar float+ CowT[Years];
083: dvar float+ Heif[Years];
084: dvar float+ Pr[Years];
085: 
086: dexpr float objective =
087:   sum(y in Years) Pr[y]
088:     - Repay * (sum(y in Years) ((LoanTerm-NbYears-1+y) * CapAdd[y]));
089: 
090: maximize objective;
091: 
092: subject to {
093:   // Add a Total cows variable
094:   forall(y in Years)
095:     ctTotCow: CowT[y] == sum(j in CowAges) Cow[j][y];
096: 
097:   // Continuity - cows move from one age to the next
098:   forall(y in 1..NbYears-1) {
099:     Cow[1][y+1] == HeifSurvival * Heif[y];
100:     Cow[2][y+1] == HeifSurvival * Cow[1][y];
101:     forall(j in CowAges)
102:       Cow[j+1][y+1] == CowSurvival * Cow[j][y];
103:   }
104: 
105:   // Sell heifers as newborns
106:   forall(y in Years)
107:     ctHeifNew:  Heif[y] == CalfRate * HeifFraction * CowT[y] - HeifSell[y];
108: 
109:   // Initial conditions
110:   Cow[1][1] == HeifSurvival * InitCows[1];
111:   Cow[2][1] == HeifSurvival * InitCows[2];
112:   forall(j in 3..NbAges)
113:     Cow[j][1] == CowSurvival * InitCows[j];
114:  
115:   // Accomodation
116:   forall(y in Years)
117:     ctAccom: Heif[y] + Cow[1][y] + CowT[y]
118:        <= InitCap + sum(k in Years: k <= y) CapAdd[k];
119: 
120:   // Grain consumption
121:   forall(y in Years)
122:     ctGrainCon: CowT[y] * GrPerCow <= 
123:        sum(l in Lands) GrGrow[l,y] + GrBuy[y] - GrSell[y];
124:           
125:   // Sugar beet consumption
126:   forall(y in Years)
127:     ctSugBeet: CowT[y] * SbPerCow <=
128:        SbGrow[y] + SbBuy[y] - SbSell[y];
129: 
130:   // Acreage
131:   forall(y in Years)
132:     ctAcreage: sum(l in Lands) 1.0/GrPerAcre[l] * GrGrow[l][y] 
133:        + 1.0/SbPerAcre * SbGrow[y] + HeifAcre * Heif[y]
134:        + HeifAcre * Cow[1][y] 
135:        + CowAcre * CowT[y] <= Acres; 
136: 
137:   // Labor
138:   forall(y in Years)
139:     ctLabor: HeifLabor * Heif[y] + HeifLabor * Cow[1][y]
140:     + CowLabor * CowT[y]
141:     + GrLabor * (sum(l in Lands) 1.0/GrPerAcre[l] * GrGrow[l][y])
142:     + SbLabor * 1.0 / SbPerAcre * SbGrow[y]
143:     <= Labor + Lbr[y];
144: 
145:   // End total
146:   ctEndTot: InitCowsTotal * (1.0 - DownCowChg) 
147:       <= CowT[NbYears]
148:       <= InitCowsTotal * (1.0 + UpCowChg);
149: 
150:   // Profit
151:   forall(y in Years)
152:     ctProfit: Pr[y] == BullockSellPrice * CalfRate * (1.0 - HeifFraction) * CowT[y] 
153:              + HeifSellPrice * HeifSell[y]
154:              + CowSellPrice * Cow[NbAges][y]
155:              + MilkPerCowPrice * CowT[y] 
156:              + GrSellPrice * GrSell[y]
157:              + SbSellPrice * SbSell[y]
158:              - GrBuyPrice * GrBuy[y]
159:              - SbBuyPrice * SbBuy[y]
160:              - LbrCost * Lbr[y]
161:              - LbrFixCost
162:              - HeifCost * Heif[y]
163:              - HeifCost * Cow[1][y]
164:              - CowCost * CowT[y]
165:              - GrGrowCost * (sum(l in Lands) 1.0/GrPerAcre[l] * GrGrow[l][y])
166:              - SbGrowCost * (1.0/SbPerAcre) * SbGrow[y]
167:              - Repay * (sum(k in Years: k<=y) CapAdd[k]);
168: 
169:   // Profit can never be negative--variable declared >= 0 above
170: }
171: 
172: execute DISPLAY {   
173:   for(var y in Years)
174:     for(var l in Lands)
175:       if(GrGrow[l][y] >= 1e-6)
176:         writeln("GrGrow[", l, "][", y, "] = ", GrGrow[l][y]);
177:             
178:   writeln();
179:

183:

189:          
190:   writeln();
191:    
192:   for(y in Years)
193:     if(SbBuy[y] >= 1e-6)
194:       writeln("SbBuy[", y, "] = ", SbBuy[y]);
195:    
196:   writeln();
197:         
198:   for(y in Years)
199:     if(GrSell[y] >= 1e-6)
200:       writeln("GrSell[", y, "] = ", GrSell[y]);
201:    
202:   writeln();
203:          
204:   for(y in Years)
205:     if(SbSell[y] >= 1e-6)
206:       writeln("SbSell[", y, "] = ", SbSell[y]);
207:    
208:   writeln();
209:          
210:   for(y in Years)
211:     if(HeifSell[y] >= 1e-6)
212:       writeln("HeifSell[", y, "] = ", HeifSell[y]);
213:    
214:   writeln();
215:          
216:   for(y in Years)
217:     writeln("Pr[", y, "] = ", Pr[y]);
218: }
