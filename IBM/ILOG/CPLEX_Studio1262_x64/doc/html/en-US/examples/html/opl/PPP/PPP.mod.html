31: using CP;
32: 
33: int numBoats = ...;
34: range Boats = 0..numBoats - 1;
35: int boatSize[Boats] = ...;
36: int crewSize[Boats] = ...;
37: 
38: int numPeriods = 6;
39: range Periods = 0..numPeriods - 1;
40: 
41: dvar boolean host[Boats];
42: dvar int goWhere[Periods][Boats] in Boats;
43: dvar int+ load[Periods][Boats];
44: 
45: dexpr int numHosts = sum (b in Boats) host[b];
46: 
47: minimize numHosts;
48: 
49: subject to {
50:    // Capacity of hosts, non-hosts have zero capacity
51:    forall (b in Boats, p in Periods)
52:      load[p][b] <= host[b] * boatSize[b];
53:    
54:    // Capacities respected
55:    forall (p in Periods)
56:      pack(all (b in Boats) load[p][b], all (b in Boats) goWhere[p][b], crewSize, numHosts);   
57: 
58:    // Hosts are always in their boat, guests are never in their boat
59:    forall (b in Boats)
60:      count(all(p in Periods) goWhere[p][b], b) == host[b] * numPeriods;
61:      
62:    // No two crews meet more than once
63:    forall (b1, b2 in Boats : b1 < b2)
64:      sum (p in Periods) (goWhere[p][b1] == goWhere[p][b2]) <= 1;
65:    
66:    // Asserted hosts and guests (in spec)
67:    host[0] == true;
68:    host[1] == true;
69:    host[2] == true;
70:    host[39] == false;
71:    host[40] == false;
72:    host[41] == false;
73: }
74: 
75: tuple hostSolutionT{ 
76:         int Boats; 
77:         int value; 
78: };
79: {hostSolutionT} hostSolution = {<i0,host[i0]> | i0 in Boats};
80: tuple loadSolutionT{ 
81:         int Periods; 
82:         int Boats; 
83:         int value; 
84: };
85: {loadSolutionT} loadSolution = {<i0,i1,load[i0][i1]> | i0 in Periods,i1 in Boats};
86: tuple goWhereSolutionT{ 
87:         int Periods; 
88:         int Boats; 
89:         int value; 
90: };
91: {goWhereSolutionT} goWhereSolution = {<i0,i1,goWhere[i0][i1]> | i0 in Periods,i1 in Boats};
