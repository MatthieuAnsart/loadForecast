019: {string} Naptha = ...;
020: {string} Resid = ...;
021: {string} Oil = ...;
022: {string} ReformProd = ...;
023: {string} CrackProd = ...;

025: {string} Fuel = ...;
026: {string} Lube = ...;
027: 
028: float DistillNaptha[Crude][Naptha] = ...;
029: float DistillOil[Crude][Oil] = ...;
030: float DistillResid[Crude][Resid] = ...;
031: 
032: float ResidProcess[Resid][Lube] = ...;
033: float ReformProcess[Naptha][ReformProd] = ...;
034: float CrackProcess[Oil][CrackProd] = ...;
035: 
036: float VaporOil[Oil] = ...;
037: float VaporResid[Resid] = ...;
038: float VaporCrkOil = ...;
039: float LimVaporJF = ...;
040: 
041: float LimCrude[Crude] = ...;
042: float LimDistill = ...;
043: float LimReform = ...;
044: float LimCrack = ...;
045: float LoLube[Lube] = ...;
046: float UpLube[Lube] = ...;
047: 
048: float OctaneNaptha[Naptha] = ...;
049: float OctaneReform[ReformProd] = ...;
050: float OctaneCG = ...;
051: float ReqOctane[Petrol] = ...;
052: float ReqRatioPetrol = ...;
053: 
054: float ReqOilFO[Oil] = ...;
055: float ReqCrkFO = ...;
056: float ReqResidFO[Resid] = ...;
057: 
058: float ProfitPetrol[Petrol] = ...;
059: float ProfitFuel[Fuel] = ...;
060: float ProfitLube[Lube] = ...;
061: 
062: dvar float+ Cr[c in Crude] in 0..LimCrude[c];
063: dvar float+ Nap[Naptha];
064: dvar float+ Napref[Naptha][ReformProd];
065: dvar float+ Napb[Naptha][Petrol];
066: dvar float+ Refb[ReformProd][Petrol];
067: dvar float+ Ref[ReformProd];
068: dvar float+ OilVar[Oil];
069: dvar float+ Oilcrk[Oil];
070: dvar float+ Oilb[Oil][Fuel];
071: dvar float+ Crk[CrackProd];
072: dvar float+ Crkg[Petrol];
073: dvar float+ Crko[Fuel];
074: dvar float+ ResidVar[Resid];
075: dvar float+ Residl[Resid][Lube];
076: dvar float+ Residbf[Resid][Fuel];
077: dvar float+ Fpf[Fuel]; 
078: dvar float+ Fpp[Petrol]; 
079: dvar float+ Fpl[l in Lube] in LoLube[l]..UpLube[l];
080: 
081: dexpr float TotalProfitFuel =
082:    sum(f in Fuel) ProfitFuel[f] * Fpf[f];
083:      
084: dexpr float TotalProfitPetrol =
085:    sum(p in Petrol) ProfitPetrol[p] * Fpp[p];
086:    
087: dexpr float TotalProfitLube = 
088:    sum(l in Lube) ProfitLube[l] * Fpl[l];
089:    
090: maximize TotalProfitFuel + TotalProfitPetrol + TotalProfitLube;
091: 
092: subject to {
093:   // Distallation capacity
094:   ctDistCap: sum(c in Crude) Cr[c] <= LimDistill;
095: 
096:   // Reforming capacity
097:   ctRefCap:
098:     sum(n in Naptha, r in ReformProd) Napref[n][r] <= LimReform;
099: 
100:   // Cracking capacity
101:   ctCrkCap: sum(o in Oil) Oilcrk[o] <= LimCrack;
102: 
103:   // Distallation products
104:   forall(n in Naptha)
105:     sum(c in Crude) DistillNaptha[c][n] * Cr[c] == Nap[n];     
106: 
107:   forall(o in Oil)
108:     sum(c in Crude) DistillOil[c][o] * Cr[c] == OilVar[o];     
109: 
110:   forall(r in Resid)
111:     sum(c in Crude) DistillResid[c][r] * Cr[c] == ResidVar[r];     
112: 
113:   // Reformer products
114:   forall (r in ReformProd)
115:     sum(n in Naptha) ReformProcess[n][r] * Napref[n][r] == Ref[r]; 
116:    
117:   // Cracking products
118:   forall(c in CrackProd)
119:     sum(o in Oil) CrackProcess[o][c] * Oilcrk[o] == Crk[c];
120: 
121:   Crk["CG"] == sum(p in Petrol) Crkg[p];
122:   Crk["CO"] == sum(f in Fuel) Crko[f];
123: 
124:   // Resid process
125:   forall(l in Lube)
126:     sum(r in Resid) ResidProcess[r][l] * Residl[r][l] == Fpl[l];
127: 
128:   // Balance constraints on Napthas
129:   forall(n in Naptha)
130:     sum(r in ReformProd) Napref[n][r] + sum(p in Petrol) Napb[n][p] == Nap[n];
131: 
132:   // Balance constraints on Oils
133:   forall(o in Oil)
134:     Oilcrk[o] + sum(f in Fuel) Oilb[o][f] == OilVar[o];   
135: 
136:   // Balance constraints on Resids
137:   forall(r in Resid)
138:     sum(f in Fuel) Residbf[r][f] + sum(l in Lube) Residl[r][l] == ResidVar[r];
139: 
140:   // Balance constaint on Reformer products
141:   forall(r in ReformProd)
142:     sum(p in Petrol) Refb[r][p] == Ref[r];
143: 
144:   // Balance constraints on Petrols
145:   forall(p in Petrol)
146:     sum(n in Naptha) Napb[n][p] + sum(r in ReformProd) Refb[r][p] + Crkg[p] == Fpp[p];
147: 
148:   // Balance constraint on Fuels
149:   forall(f in Fuel)
150:     sum(o in Oil) Oilb[o][f] + Crko[f] + sum(r in Resid) Residbf[r][f] == Fpf[f];
151:    
152:   // Fixed proportions required for Fuel Oil
153:   forall(o in Oil)
154:     Oilb[o,"FO"] == ReqOilFO[o] * Fpf["FO"];
155:   Crko["FO"] == ReqCrkFO * Fpf["FO"];
156:   forall(r in Resid)
157:     Residbf[r]["FO"] == ReqResidFO[r] * Fpf["FO"]; 
158: 
159:   // Required ratio beteen petrols
160:   cttReqRatio: Fpp["PMF"] >= ReqRatioPetrol * Fpp["RMF"];
161:    
162:   // Qualities
163:   // Octane
164:   forall(p in Petrol)
165:     ctOctane :      
166:       sum(n in Naptha) OctaneNaptha[n] * Napb[n][p] + 
167:       sum(r in ReformProd) OctaneReform[r] * Refb[r][p]
168:          + OctaneCG * Crkg[p] >= ReqOctane[p] * Fpp[p]; 
169: 
170:   // Vapor Pressure constraint on Jet Fuel
171:   ctVapPres: 
172:   sum(o in Oil) VaporOil[o] * Oilb[o]["JF"] + VaporCrkOil * Crko["JF"]
173:        + sum(r in Resid) VaporResid[r] * Residbf[r]["JF"] <= LimVaporJF * Fpf["JF"];
174:       
175: }
