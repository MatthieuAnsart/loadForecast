025: 
026: int      Horizon = 60*24*7-1;    // Number of minutes in a week
027: int      MaxRefuel = 20;         
028: {string} Airports = ...;
029: {string} Fleets = ...;
030: {string} Distance = ...;
031: float    MaxSpill = ...;
032: 
033: //Characteristics of each fleet
034: tuple fleetType 
035: {
036:   int      aircrafts;  //number of aircrafts for each fleet
037:   string   dist;       //long-medium-short distance flight
038:   int      seats;      //number of seats
039:   int      refuelT;    //Time on ground between flights
040:   int      a;          //fixed cost
041:   int      b;          //variable cost : total cost = a + BH * b * seats where BH is the flight time
042: }
043: fleetType FleetInfo[Fleets] = ...; 
044: assert forall(pl in Fleets) FleetInfo[pl].refuelT <= MaxRefuel;
045: 
046: //Flight legs
047: tuple flightLegKey 
048: {
049:   int       id;    //flight id
050:   string    depA;  //departure airport
051:   int       depT;  //departure time 
052:   string    arrA;  //arrival airport
053:   int       arrT;  //arrival time 
054:   string    dist;  //long-medium-short distance flight
055:   int       pax;   //passenger demand
056:   int       price; //ticket price
057: }
058: {flightLegKey} FlightLegs = ...; 
059: assert forall(fl in FlightLegs)  0 <= fl.depT;// <= Horizon;
060: assert forall(fl in FlightLegs)  0 <= fl.arrT;// <= Horizon;
061: assert forall(fl in FlightLegs)  Horizon >= fl.depT;
062: assert forall(fl in FlightLegs)  Horizon >= fl.arrT;
063: assert forall(fl in FlightLegs)  fl.depT < fl.arrT;
064: 
065: tuple flightLeg 
066: {
067:   int       id;    //flight id
068:   string    depA;  //departure airport
069:   int       depT;  //departure time 
070:   string    arrA;  //arrival airport
071:   int       arrT;  //arrival time 
072:   string    dist;  //long-medium-short distance flight
073:   int       pax;   //passenger demand
074:   int       price; //ticket price
075: }
076: //Flow model
077: {flightLeg} Source = {<999,"Source", f.depT-MaxRefuel,f.depA, f.depT-MaxRefuel,f.dist,0,0> | f in FlightLegs};
078: tuple flightLegKey2 
079: {
080:   int       id;    //flight id
081:   key string    depA;  //departure airport
082:   int       depT;  //departure time 
083:   string    arrA;  //arrival airport
084:   int       arrT;  //arrival time 
085:   key string    dist;  //long-medium-short distance flight
086:   int       pax;   //passenger demand
087:   int       price; //ticket price
088: }
089: 
090: {flightLegKey2} Sink = {<999,f.arrA, Horizon+MaxRefuel, "Sink", Horizon+MaxRefuel,f.dist,0,0> | f in FlightLegs};
091: {flightLeg} Flights = FlightLegs union Source union Sink;
092: 
093: //One-stop flights
094: //These are flights which are broken into 2 sub-flights, and are flown by the
095: //same aircraft.
096: tuple oneStop
097: {
098:   key int firstId;
099:   int secondId;
100: }
101: {oneStop} OneStopFlights = ...;
102: 
103: //Cash Direct Operating Costs
104: //for every couple (fleet,flight) we define  cdoc = a + b*(demand-spill)*(arrTime-depTime)
105: int Cost[Flights][Fleets];
106: 
107: //Profit
108: //for every couple (fleet,flight) we define  profit = (demand-spill)*ticket_price
109: int Profit[Flights][Fleets];
110: 
111: //Cost & Profit Initialization
112: execute INITIALIZE {
113:    for(var fl in Flights) {
114:       for(var pl in Fleets) {
115:          if(fl.depA != "Source" && fl.arrA != "Sink") {  
116:             //Cost vector
117:             if(fl.pax > FleetInfo[pl].seats) {
118:                Cost[fl][pl] = FleetInfo[pl].a + FleetInfo[pl].b*FleetInfo[pl].seats*(fl.arrT-fl.depT);
119:             } else {
120:                Cost[fl][pl] = FleetInfo[pl].a + FleetInfo[pl].b*fl.pax*(fl.arrT-fl.depT);
121:             }
122: 
123:             //Profit vector
124:             if(fl.pax > FleetInfo[pl].seats) {
125:                Profit[fl][pl] = FleetInfo[pl].seats*fl.price;
126:             } else {
127:                Profit[fl][pl] = fl.pax*fl.price;
128:             }
129:          } else {
130:             Cost[fl][pl] = 0;
131:             Profit[fl][pl] = 0;
132:          }
133:       }
134:    }
135: }
136: 
137: 
138: /*  ----------------------------------------------------
139:  *   Variables:
140:  *   assignment -- assignment[fl][pl] means flights[fl] is 
141:  *         covered by a plane in fleet[pl].
142:  *   --------------------------------------------------- */
143: dvar boolean Assignment[Flights][Fleets];
144: 
145: dexpr float objective = 
146:   sum(fl in Flights, pl in Fleets)
147:      (-Profit[fl][pl] + Cost[fl][pl]) * Assignment[fl][pl];
148:       
149: minimize objective;
150: 
151: subject to {
152:   // Every plane of each fleet must come from the source only once.
153:   forall(pl in Fleets)
154:     ctSource: sum(fl in Flights: fl.depA == "Source") Assignment[fl][pl] <= FleetInfo[pl].aircrafts;
155: 
156:   // Every plane of each fleet must go to the sink only once.
157:   forall(pl in Fleets)
158:     ctSink: sum(fl in Flights: fl.arrA == "Sink") Assignment[fl][pl] <= FleetInfo[pl].aircrafts;
159: 
160:   forall(fl in Flights: fl.depA != "Source" && fl.arrA != "Sink") 
161:     // Every "real" flight must have a plane assigned to it
162:     sum(pl in Fleets) Assignment[fl][pl] == 1;
163: 
164:   forall(pl in Fleets, fl in Flights: fl.depA != "Source" && fl.arrA != "Sink") 
165:     // The plane must be at the airport in order to use it!
166:     sum(prevf in Flights: prevf.arrA == fl.depA && prevf.arrT + FleetInfo[pl].refuelT <= fl.depT) 
167:       Assignment[prevf][pl] -
168:     sum(prevf in Flights: prevf != fl && prevf.depA == fl.depA && prevf.depT <= fl.depT) 
169:       Assignment[prevf][pl]
170:     >= Assignment[fl][pl];
171:       
172:   forall(pl in Fleets, fl in Flights: fl.arrA == "Sink")
173:     // The plane must be at the airport in order to use it!
174:     sum(prevf in Flights: prevf.arrA == fl.depA && prevf.arrT + FleetInfo[pl].refuelT <= fl.depT) 
175:       Assignment[prevf][pl] -
176:     sum(prevf in Flights: prevf != fl && prevf.depA == fl.depA && prevf.depT <= fl.depT) 
177:       Assignment[prevf][pl]
178:     >= Assignment[fl][pl];
179:      
180:   //Type-compatibility check between fleets and flights (long/medium/short distance)
181:   //Long-haul aircrafts can fly long, medium and short flights, medium-haul aircrafts
182:   //can only fly medium and short flights, and so on..
183:   forall(fl in Flights: fl.depA != "Source" && fl.arrA != "Sink")
184:     forall(pl in Fleets: FleetInfo[pl].dist == "Short" && (fl.dist == "Medium" || fl.dist == "Long"))
185:       Assignment[fl][pl] == 0;
186:   forall(fl in Flights: fl.depA != "Source" && fl.arrA != "Sink")
187:     forall(pl in Fleets: FleetInfo[pl].dist == "Medium" && fl.dist == "Long")
188:       Assignment[fl][pl] == 0;
189: 
190:   forall(fl in Flights: fl.depA != "Source" && fl.arrA != "Sink")
191:     //MaxSpill constraint
192:     //if MaxSpill=.1 and the demand for the flight f is 100, then the maximum no of passengers 
193:     //that can be spilled is 10 (i.e. you must use an aircraft which has seat capacity >= 90 )
194:     ctSpill: sum(pl in Fleets) Assignment[fl][pl]*FleetInfo[pl].seats >= fl.pax*(1-MaxSpill);
195: 
196:   forall(pl in Fleets) {
197:     // Planes must end the day where they started the day
198:     forall(ap in Airports) {
199:       sum(fl in Flights: fl.depA == "Source" && fl.arrA == ap) Assignment[fl][pl] ==
200:       sum(fl in Flights: fl.depA == ap && fl.arrA == "Sink") Assignment[fl][pl];
201:     }
202: 
203:     //One-Stop services
204:     //force each ordered pair of forced turn flights to use the same equipment type
205:     forall(pair in OneStopFlights) {
206:       sum(f1 in Flights: f1.id == pair.firstId) Assignment[f1][pl] ==
207:       sum(f2 in Flights: f2.id == pair.secondId) Assignment[f2][pl];
208:     }
209:   }
210: }
211: 
212: tuple AssignmentSolutionT{ 
213:         flightLegKey Flights;   
214:         string Fleets; 
215:         int value; 
216: };
217: {AssignmentSolutionT} AssignmentSolution = {<i0,i1,Assignment[i0][i1]> | i0 in Flights,i1 in Fleets};
218: 
219: 
220: execute DISPLAY {
221:    for(var fl in Flights)
222:       for(var pl in Fleets)
223:          if(Assignment[fl][pl] == 1) 
224:             writeln("Assignment[", fl, "][", pl, "] = ", Assignment[fl][pl]);
225: }
