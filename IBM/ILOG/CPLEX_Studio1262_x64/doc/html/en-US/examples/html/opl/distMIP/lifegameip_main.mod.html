25: main {
26:   var m1Source = new IloOplModelSource("lifegameip.mod");
27:   var def = new IloOplModelDefinition(m1Source);
28:   var cplex1 = new IloCplex();
29:   cplex1.readVMConfig(thisOplModel.resolvePath("process.vmc"));
30: 
31:   var opl1 = new IloOplModel(def, cplex1);
32:   opl1.generate();
33:   cplex1.solve();
34:   writeln(opl1.printSolution());
35: 
36:   if (cplex1.hasVMConfig()) {
37:     writeln("cplex1 has a VM config");
38:   } else {
39:     fail();
40:   }
41: 
42:   cplex1.delVMConfig();
43:   if (cplex1.hasVMConfig()) {
44:     writeln("cplex1 has a VM config");
45:   } else {
46:     writeln("cplex1 does not have anymore a VM config");
47:   }
48: }
