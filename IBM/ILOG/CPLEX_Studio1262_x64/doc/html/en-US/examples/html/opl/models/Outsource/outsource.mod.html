020: int NumAmts = ...; // # of discount ranges
021: {string} Items = ...;
022: {string} Suppliers = ...;
023: {string} AUDsuppliers = ...; // AUD (All-Units-Discount) suppliers
024: {string} CQDsuppliers = ...; // ICD (Incremental-Quantity-Discount) suppliers 
025: 
026: assert card(CQDsuppliers union AUDsuppliers) == card(Suppliers);
027: 
028: range Amts = 1..NumAmts; // Indices of the discount ranges
029: 
030: int ItemDemand[Items] = ...; 
031: int TotalSupplierCap[Suppliers] = ...; // Supply capacity for all items at each supplier
032: int BreakAmts[1..NumAmts+1] = ...; // Discount Breaks
033: int SetupCost[Suppliers] = ...;
034: 
035: //float+ basicCost[Items][Suppliers] = ...;
036: float CostInRanges[Items][Suppliers][Amts] = ...; //Cost difference between consecutive discount ranges
037: 
038: int MaxN = sum(i in Suppliers) TotalSupplierCap[i];
039: 
040: 
041: dvar int Quantity[Items][Suppliers][Amts] in 0..MaxN;
042: dvar int SupAmt[Items][Suppliers][Amts] in 0..1; // For AUD
043: dvar int TotalQuantity[Items][Suppliers] in 0..MaxN;
044: dvar int Setup[Suppliers] in 0..1;
045: 
046: dexpr float TotalVariableCost =
047:   sum(i in Items, s in Suppliers, a in Amts) CostInRanges[i][s][a] 
048:                                                  * Quantity[i][s][a] ;
049: dexpr float TotalSetupCost =
050:   sum(s in Suppliers) SetupCost[s]*Setup[s];                                            
051:   
052: minimize TotalVariableCost + TotalSetupCost;
053:                                                  
054: subject to {
055:   // Capacity Constraints
056:   forall(s in Suppliers)
057:     ctCaps: sum(i in Items, a in Amts) Quantity[i][s][a] <= TotalSupplierCap[s];
058:        
059:   // Satisfy Demand
060:   forall(i in Items)
061:     ctDem: sum(s in Suppliers, a in Amts) Quantity[i][s][a] >= ItemDemand[i];
062:       
063:   // supAmt in one range only for each item and each All-Unit-Discount supplier
064:   forall(i in Items, s in AUDsuppliers)
065:     sum(a in Amts) SupAmt[i][s][a] == 1;
066:         
067:   // Force AUD quantity to be in correct range
068:   forall(i in Items, s in AUDsuppliers, a in Amts) {
069:     Quantity[i][s][a] <= (BreakAmts[a+1]-1) * SupAmt[i][s][a];
070:     Quantity[i][s][a] >= (BreakAmts[a]) * SupAmt[i][s][a];
071:   }
072:    
073:   //Setup indicator variables: If a supplier orders any positive quantity of any items, 
074:   // a setup cost occurs.
075:   forall(s in Suppliers) 
076:     ctSetup: Setup[s]*MaxN >= sum(i in Items, a in Amts) Quantity[i][s][a];
077: 
078:   //Force quantity in discount range at CQD supplier to be incremental 
079:   forall(i in Items, s in CQDsuppliers, a in Amts) {
080:     forall(k in 1..a-1) {
081:       // Because the "quantity" for CQDs is incremental, if CQD order quantity lies 
082:       // in discount interval a, namely, sup[i,s,a]=1, then
083:       // the quantities in interval 1 to a-1, should be the length of those ranges.  
084:       Quantity[i][s][k] >= (BreakAmts[k+1]-BreakAmts[k])*SupAmt[i][s][a];
085:     }
086:     // quantity in each range is no greater than the width of the range      
087:     Quantity[i][s][a] <= (BreakAmts[a+1]-BreakAmts[a])*SupAmt[i][s][a];             
088:   }
089: 
090:   // Total Quantity ordered for each item from each supplier
091:   forall(i in Items, s in Suppliers)
092:     sum(a in Amts) Quantity[i][s][a] == TotalQuantity[i][s];
093:       
094: }
095: 
096: 
097: tuple QuantitySolutionT{ 
098:         string Items; 
099:         string Suppliers; 
100:         int Amts; 
101:         int value; 
102: };
103: {QuantitySolutionT} QuantitySolution = {<i0,i1,i2,Quantity[i0][i1][i2]> | i0 in Items,i1 in Suppliers,i2 in Amts};
104: tuple SetupSolutionT{ 
105:         string Suppliers; 
106:         int value; 
107: };
108: {SetupSolutionT} SetupSolution = {<i0,Setup[i0]> | i0 in Suppliers};
109: tuple SupAmtSolutionT{ 
110:         string Items; 
111:         string Suppliers; 
112:         int Amts; 
113:         int value; 
114: };
115: {SupAmtSolutionT} SupAmtSolution = {<i0,i1,i2,SupAmt[i0][i1][i2]> | i0 in Items,i1 in Suppliers,i2 in Amts};
116: tuple TotalQuantitySolutionT{ 
117:         string Items; 
118:         string Suppliers; 
119:         int value; 
120: };
121: {TotalQuantitySolutionT} TotalQuantitySolution = {<i0,i1,TotalQuantity[i0][i1]> | i0 in Items,i1 in Suppliers};

124:    writeln("TotalQuantity = ", TotalQuantity);
