12: /***************************************************
13: OPL Model for Car Sequencing
14: 
15: Cars in production are placed on an
16: assembly line moving through various
17: units that install options such as air
18: conditioning and radios.  The
19: assembly line can thus be viewed as
20: composed of slots and each car must
21: be allocated to a single slot.    The
22: cars cannot be allocated arbitrarily,
23: since the production units have
24: limited capacity and the options must
25: be added to the cars as the assembly
26: line is moving in front of the unit.
27: 
28: Two projects are given, with two different
29: data sets.  The "Small" run configuration has the data
30: file smallcar.dat that will run with the
31: OPL Studio trial version.
32: ***************************************************/
33: using CP;
34: 
35: 
36: execute{
37:         }
38: 
39: int  nbCars    = ...; // # of cars
40: int   nbOptions = ...;// # of options
41: int   nbSlots   = ...;// # of slots
42: 
43: range   Cars    = 1..nbCars; 
44: range   Options = 1..nbOptions;
45: range   Slots   = 1..nbSlots;
46: 
47: int demand[Cars] = ...;
48: int option[Options,Cars] = ...; 
49: 
50: tuple Tcapacity {
51:    int l;
52:    int u;
53: };
54: Tcapacity capacity[Options] = ...; 
55: int optionDemand[i in Options] = sum(j in Cars) demand[j] * option[i,j];
56: 
57: dvar int slot[Slots] in Cars;
58: dvar int setup[Options,Slots] in 0..1;
59: 
60: subject to {
61:    // # of cars = demand
62:    forall(c in Cars )
63:       sum(s in Slots ) (slot[s] == c) == demand[c];
64:    
65:    forall(o in Options, s in 1..(nbSlots - capacity[o].u + 1) )
66:       sum(j in s..(s + capacity[o].u - 1)) setup[o,j] <= capacity[o].l;
67: 
68:    forall(o in Options, s in Slots )
69:      setup[o,s] == option[o][slot[s]];
70:    
71:    forall(o in Options, i in 1..optionDemand[o])
72:      sum(s in 1 .. (nbSlots - i * capacity[o].u)) setup[o,s] >=
73:      optionDemand[o] - i * capacity[o].l;
74: };
75: 
76: tuple slotSolutionT{ 
77:         int Slots; 
78:         int value; 
79: };
80: {slotSolutionT} slotSolution = {<i0,slot[i0]> | i0 in Slots};
81: tuple setupSolutionT{ 
82:         int Options; 
83:         int Slots; 
84:         int value; 
85: };
86: {setupSolutionT} setupSolution = {<i0,i1,setup[i0][i1]> | i0 in Options,i1 in Slots};
