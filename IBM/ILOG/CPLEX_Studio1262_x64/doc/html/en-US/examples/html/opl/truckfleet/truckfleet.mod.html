012: /* ------------------------------------------------------------
013: 
014: Problem Description
015: -------------------
016: The problem is to deliver some orders to several clients with a single truck. 
017: Each order consists of a given quantity of a product of a certain type (called
018: its color).
019: The truck must be configured in order to handle one, two or three different colors of products.
020: The cost for configuring the truck from a configuration A to a configuration B depends on A and B.
021: The configuration of the truck determines its capacity and its loading cost. 
022: A truck can only be loaded with orders for the same customer.
023: Both the cost (for configuring and loading the truck) and the number of travels needed to deliver all the 
024: orders must be minimized, the cost being the most important criterion. 
025: 
026: ------------------------------------------------------------ */
027: using CP;
028: 
029: int nbTrucks       = ...; //max number of travels of the truck
030: range Trucks       = 1..nbTrucks;
031: 
032: tuple Order {
033:   key int id;
034:   int customerId;
035:   int volume;
036:   string color;
037: }
038: {Order} Orders = ...;
039: int volumes[o in Orders] = o.volume;
040: {int} Customers = { o.customerId | o in Orders };
041: 
042: tuple TruckConfig {
043:   key int id;
044:   int maxLoad;
045:   int cost;
046:   {string} allowedColors;
047: }
048: {TruckConfig} TruckConfigs = ...;
049: {int} TruckConfigIds = { t.id | t in TruckConfigs};
050: int minTruckConfigId = min(i in TruckConfigIds) i; 
051: int maxTruckConfigId = max(i in TruckConfigIds) i; 
052: range TruckConfigRange = minTruckConfigId..maxTruckConfigId;
053: int maxTruckConfigLoad[TruckConfigIds] = [t.id : t.maxLoad | t in TruckConfigs];
054: int maxLoad = max(t in TruckConfigIds) maxTruckConfigLoad[t]; 
055: 
056: //cost for loading a truck of a given config
057: int truckCost[TruckConfigIds] =  [t.id : t.cost | t in TruckConfigs];
058: 
059: //configuration of the truck
060: dvar int truckConfigs[Trucks] in TruckConfigRange;
061: //In which truck is an order
062: dvar int where[Orders] in Trucks;
063: //load of a truck
064: dvar int load[Trucks] in 0..maxLoad;
065: // number of trucks used
066: dvar int numUsed in 0..nbTrucks;
067: dvar int customerOfTruck[Trucks] in (min(i in Customers) i)..(max(i in Customers) i);
068: 
069: tuple Transition {
070:    int src;
071:    int dest;
072:    int cost;
073: };
074: {Transition} costTuples = ...;
075: dvar int transitionCost[Trucks] in 0..(max(t in costTuples) t.cost);
076: dvar int configOfContainer[Orders] in TruckConfigRange;
077: 
078: execute {
079:   cp.param.timeLimit=20;
080:   cp.param.logPeriod=50000;
081:   cp.setSearchPhases(cp.factory.searchPhase(where));
082: }
083: 
084: // Objective: first criterion for minimizing the cost for configuring and loading trucks 
085: //            second criterion for minimizing the number of trucks
086: dexpr int e1 =   sum(t in Trucks) (truckCost[truckConfigs[t]]*(load[t]!=0))
087:   + sum(t in Trucks) transitionCost[t];
088: dexpr int e2 = numUsed;
089: 
090: minimize staticLex(e1, e2);  // trying to minimize cost first
091: //minimize staticLex(e2, e1);  // trying to minimize numUsed first
092: subject to {
093:   forall(t in Trucks)
094:     truckConfigs[t] in TruckConfigIds;
095:   forall(o in Orders)
096:     configOfContainer[o] in TruckConfigIds;
097:   forall(t in Trucks)
098:     customerOfTruck[t] in Customers;
099: 
100:   forall(t in 2..nbTrucks)
101:     allowedAssignments(costTuples, truckConfigs[t-1], truckConfigs[t], transitionCost[t-1]);
102: 
103:   // constrain the volume of the orders in each truck 
104:   pack(load, where, volumes, numUsed);
105:   forall(t in Trucks)
106:     load[t] <= maxTruckConfigLoad[truckConfigs[t]];
107: 
108:   // compatibility between the colors of an order and the configuration of its truck 
109:   forall(o in Orders) {
110:     //configOfContainer[o] in allowedContainerConfigs[o.color];
111:     configOfContainer[o] in { t.id | t in TruckConfigs : o.color in t.allowedColors};
112:     configOfContainer[o] == truckConfigs[where[o]];
113:   }
114: 
115:   // only one customer per truck 
116:   forall(o in Orders)
117:     customerOfTruck[where[o]] == o.customerId;
118: 
119:   // non used trucks are at the end
120:   forall(t in 2..nbTrucks)
121:     (load[t-1]) > 0 || (load[t] == 0);
122: 
123:   // Dominance: the non used truck keep the last used configuration
124:   load[1] > 0;
125:   forall(t in 2..nbTrucks)
126:     (load[t] > 0) || (truckConfigs[t] == truckConfigs[t-1]);
127: 
128:   //Dominance:  regroup deliveries with same configuration
129:   forall(t in 2..nbTrucks-1)
130:     (truckConfigs[t] == truckConfigs[t-1])
131:     || and(p in (t+1)..nbTrucks) truckConfigs[p] != truckConfigs[t-1];
132: }
133: 
134: tuple solutionT{
135:         int truck;
136:         int config;
137:         Order o;
138: };
139: {solutionT} solution = {<t, truckConfigs[t], o> | t in Trucks : load[t]!=0, o in Orders : where[o] == t};
140: 
141: execute {
142:   writeln("Configuration cost: " + e1 +
143:           " Number of Trucks: " + numUsed);
144:           
145:   for(var t in Trucks) {  
146:     if (load[t]!=0) {
147:       write("Truck " + t + ": Config=" + truckConfigs[t] + " Items= ");
148:       for (var o in Orders) {
149:         if (where[o] == t) {
150:           write("<" + o.id + "," + o.color + "," + o.volume + "> ");
151:         }
152:       }
153:       writeln();
154:     }
155:   }
156: };
