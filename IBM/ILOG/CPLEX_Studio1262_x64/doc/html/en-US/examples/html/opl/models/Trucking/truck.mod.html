015:

021: 
022: {string} Location = ...;
023: {string} TruckTypes = ...;
024: 
025: {string} Spokes = ...;
026: {string} Hubs = ...;
027: 
028: tuple spokeInfo {
029:   int     minDepTime; // Earliest departure time at spoke 
030:   int     maxArrTime; // Latest arrive time at spoke
031: };   
032: 
033: spokeInfo Spoke[Spokes] = ...;
034: 
035: // Make sure the data is consistent: latest arrive time >= earliest departure time
036: assert forall(s in Spokes) Spoke[s].maxArrTime > Spoke[s].minDepTime;
037: 
038: tuple truckTypeInfo {
039:   int   capacity;
040:   int   costPerMile;  
041:   int   milesPerHour; //speed
042: }
043: 
044: truckTypeInfo TruckTypeInfos[TruckTypes] = ...;
045: int LoadTime[Hubs][TruckTypes] = ...; // in minutes; loadTime = unloadTime
046: 
047: tuple routeInfo {
048:   key string   spoke;
049:   key string   hub;
050:   int      distance;  // in miles
051: }
052: {routeInfo} Routes = ...;
053: 
054: // The following assertion is to make sure that the spoke 
055: // in each route is indeed in the set of Spokes.
056: assert forall(r in Routes : r.spoke not in Spokes) 1 == 0;
057: 
058: // The following assertion is to make sure that the hub
059: // in each route are indeed in the set of Hubs.
060: assert forall(r in Routes : r.hub not in Hubs) 1 == 0;
061: 
062: tuple triple {
063:   string origin;
064:   string hub;
065:   string destination;
066: }
067: 
068: {triple} Triples =  // feasible pathes from spokes to spokes via one hub  
069: {<r1.spoke,r1.hub,r2.spoke> | r1,r2 in Routes : r1 != r2 && r1.hub == r2.hub};
070:  
071: tuple shipment {
072:   key string   origin;
073:   key string   destination;
074:   int       totalVolume;
075: }
076: {shipment} Shipments = ...;
077: 
078: // The following assertion is to make sure that the origin 
079: // of each shipment is indeed in the set of Spokes.
080: assert forall(s in Shipments : s.origin not in Spokes) 1 == 0;
081: 
082: // The following assertion is to make sure that the destination 
083: // of each shipment is indeed in the set of Spokes.
084: assert forall(s in Shipments : s.destination not in Spokes) 1 == 0;
085: 
086: int PossibleTruckOnRoute[Routes][TruckTypes];
087: 
088: // the earliest unloading time at a hub for each type of trucks  
089: int EarliestUnloadingTime[Routes][TruckTypes]; 
090: // the latest loading time at a hub for each type of trucks
091: int LatestLoadingTime[Routes][TruckTypes];
092: 
093: // Compute possible truck types that can be assigned on a route
094: execute INITIALIZE {
095:   for(var r in Routes)
096:     for(var t in TruckTypes) {
097:       EarliestUnloadingTime[r][t] = Math.ceil(LoadTime[r.hub][t] + Spoke[r.spoke].minDepTime + 60*r.distance/TruckTypeInfos[t].milesPerHour);
098:       LatestLoadingTime[r][t] = Math.floor(Spoke[r.spoke].maxArrTime - LoadTime[r.hub][t] - 60*r.distance/TruckTypeInfos[t].milesPerHour);
099:       // A type of truck can be assigned on a route only if it can make it to the hub and back 
100:       //  before the max arrival time at the spoke.
101:       if ( EarliestUnloadingTime[r][t] < LatestLoadingTime[r][t]) { 
102:         PossibleTruckOnRoute[r][t] = 1;
103:       }
104:       else {
105:         PossibleTruckOnRoute[r][t] = 0;
106:       }   
107:     }
108: }
109: 
110: int MaxTrucks = 100;  // Maximum # of trucks for each type on a route 
111: 
112: // Maximum Volume of goods that can be handled 
113: // on each path for each type of trucks
114: int MaxVolume = 5000; 
115: 
116: dvar int+ TruckOnRoute[Routes][TruckTypes] in 0..MaxTrucks;
117: 
118: // This represents the volumes shipped out from each hub 
119: // by each type of trucks on each triple
120: // The volumes are distinguished by trucktypes because trucks of different types 
121: // arrive at a hub at different times and the timing is used in defining 
122: // the constraints for volume availability for the trucks leaving the hub. 
123: dvar int+ OutVolumeThroughHubOnTruck[Triples][TruckTypes] in 0..MaxVolume;
124: 
125: // This represents the volume shipped into each hub by each type of trucks on each triple
126: // It is used in defining timing constraints. 
127: dvar int+ InVolumeThroughHubOnTruck[Triples][TruckTypes] in 0..MaxVolume;
128: 
129: dexpr float TotalCost = 
130:   sum(r in Routes, t in TruckTypes) 2*r.distance*TruckTypeInfos[t].costPerMile*TruckOnRoute[r][t];
131:   
132: minimize TotalCost;
133: 
134: subject to {      
135:   // The # of trucks of each type should be less than "maxTrucks", and if a type of truck 
136:   // is impossible for a route, its # should be zero 
137:   forall(r in Routes, t in TruckTypes)
138:     ctMaxTruck: 
139:       TruckOnRoute[r][t] <= PossibleTruckOnRoute[r][t]*MaxTrucks;
140: 
141:   // On each route s-h, the total inbound volume carried by trucks of each type 
142:   // should be less than the total capacity of the trucks of this type.
143:   forall(<s,h,dist> in Routes, t in TruckTypes)
144:     ctInCapacity: 
145:       sum(<s,h,dest> in Triples) InVolumeThroughHubOnTruck[<s,h,dest>][t] 
146:          <= TruckOnRoute[<s,h,dist>][t]*TruckTypeInfos[t].capacity;
147:          
148:   // On each route s-h, the total outbound volume carried by each truck type should be less than 
149:   // the total capacity of this type of truck.
150: 
151:   forall(<s,h,dist> in Routes, t in TruckTypes)
152:     ctOutCapacity:      
153:       sum(<o,h,s> in Triples) OutVolumeThroughHubOnTruck[<o,h,s>][t]
154:            <= TruckOnRoute[<s,h,dist>][t]*TruckTypeInfos[t].capacity;
155:    
156:   // On any triple, the total flows in the hub = the total flows out the hub
157:   forall (tr in Triples) 
158:     ctFlow: 
159:       sum(t in TruckTypes) InVolumeThroughHubOnTruck[tr][t]
160:         == sum(t in TruckTypes) OutVolumeThroughHubOnTruck[tr][t];
161:    
162:   // The sum of flows between any origin-destination pair via all hubs is equal to the shipment between the o-d pair.
163: 
164:   forall (<o,d,v> in Shipments )
165:     ctOrigDest: 
166:       sum(t in TruckTypes, <o,h,d> in Triples) InVolumeThroughHubOnTruck[<o,h,d>][t] == v;
167:    
168:           
169:   // There must be enough volume for a truck before it leaves a hub. 
170:   // In another words, the shipments for a truck must arrive 
171:   // at the hub from all spokes before the truck leaves.
172:   // The constraint can be expressed as the following:
173:   // For each route s-h and leaving truck of type t:
174:   // Cumulated inbound volume arrived before the loading time of the truck >=
175:   // Cumulated outbound volume upto the loading time of the truck(including the shipments being loaded).    
176:   forall (<s,h,dist> in Routes, t in TruckTypes)  
177:     ctTiming: 
178:       sum (<o,h,s> in Triples, t1 in TruckTypes, <o,h,dist1> in Routes :
179:           // The expression below defines the indices of the trucks unloaded before truck t starts loading.  
180:           EarliestUnloadingTime[<o,h,dist1>][t1] <= LatestLoadingTime[<s,h,dist>][t]) 
181:           InVolumeThroughHubOnTruck[<o,h,s>][t1] >=
182:       sum (<o,h,s> in Triples, t2 in TruckTypes, <o,h,dist2> in Routes : 
183:           // The expression below defines the indices of the trucks left before truck t starts loading.
184:           LatestLoadingTime[<o,h,dist2>][t2] <= LatestLoadingTime[<s,h,dist>][t]) 
185:           OutVolumeThroughHubOnTruck[<o,h,s>][t2];
186: }
187: 
188: 
189: /************************************************************
190:        POST-PROCESSING                                    
191: *************************************************************/
192: // Post processing: result data structures are exported as post-processed tuple or tuple sets
193: // Solve objective value
194: tuple result {
195:   float totalCost;
196: }
197: result Result = <TotalCost>;
198: 
199: // Number of trucks assigned to each route, for each truck type
200: tuple nbTrucksOnRouteRes {
201:   key string    spoke;
202:   key string    hub;
203:   key string    truckType;
204:   int                   nbTruck;
205: }
206: {nbTrucksOnRouteRes} NbTrucksOnRouteRes = {<r.spoke, r.hub, t, TruckOnRoute[r][t]> | r in Routes, t in TruckTypes};
207: 
208: // Volume shipped into each hub by each type of trucks and each pair (origin, destination)
209: tuple inVolumeThroughHubOnTruckRes {
210:   key string    origin;
211:   key string    hub;
212:   key string    destination;
213:   key string    truckType;
214:   int                   quantity;
215: }
216: {inVolumeThroughHubOnTruckRes} InVolumeThroughHubOnTruckRes =
217:         {<tr.origin, tr.hub, tr.destination, t, InVolumeThroughHubOnTruck[tr][t]> | tr in Triples, t in TruckTypes};
218: 
219: // Volume shipped from each hub by each type of trucks and each pair (origin, destination)
220: tuple outVolumeThroughHubOnTruckRes {
221:   key string    origin;
222:   key string    hub;
223:   key string    destination;
224:   key string    truckType;
225:   int                   quantity;
226: }
227: {outVolumeThroughHubOnTruckRes} OutVolumeThroughHubOnTruckRes =
228:         {<tr.origin, tr.hub, tr.destination, t, OutVolumeThroughHubOnTruck[tr][t]> | tr in Triples, t in TruckTypes};
229: 
230: execute {
231:   Result;
232:   NbTrucksOnRouteRes;
233:   InVolumeThroughHubOnTruckRes;
234:   OutVolumeThroughHubOnTruckRes;
235: }
