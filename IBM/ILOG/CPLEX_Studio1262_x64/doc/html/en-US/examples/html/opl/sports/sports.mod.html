012: using CP;
013: 
014: /* ------------------------------------------------------------
015: 
016: Problem Description
017: -------------------
018: 
019: The problem involves finding a schedule for a sports league. The league has 10 
020: teams that play games over a season of 18 weeks. Each team has a home arena and 
021: plays each other team twice during the season, once in its home arena and once in 
022: the opposing team's home arena. For each of these games, the team playing at its 
023: home arena is referred to as the home team; the team playing at the opponent's 
024: arena is called the away team. There are 90 games altogether.
025: 
026: Each of the 18 weeks in the season has five identical slots to which games can be 
027: assigned. Each team plays once a week. For each pair of teams, these two teams are 
028: opponents twice in a season; these two games must be scheduled in different halves 
029: of the season. Moreover, these two games must be scheduled at least six weeks 
030: apart. A team must play at home either the first or last week but not both.
031: 
032: A break is a sequence of consecutive weeks in which a team plays its games either 
033: all at home or all away. No team can have a break of three or more weeks in it. The
034: objective in this problem is to minimize the total number of breaks the teams play. 
035: 
036: ------------------------------------------------------------ */
037: 
038: int n = 10;
039: 
040: 
041: assert(n%2 == 0);
042: 
043: int nbWeeks = 2 * (n - 1);
044: int nbGamesPerWeek = n div 2;
045: int nbGames = n * (n - 1);
046: float mid = nbWeeks / 2 + 1;
047: int overlap = (n>=6) ? minl(n div 2, 6) : 0;
048: 
049: 
050: dvar int games[1..nbWeeks][1..nbGamesPerWeek] in 1..nbGames;
051: dvar int home[1..nbWeeks][1..nbGamesPerWeek] in 1..n;
052: dvar int away[1..nbWeeks][1..nbGamesPerWeek] in 1..n;
053: dvar int weekOfGame[1..nbGames] in  1..nbWeeks;
054: dvar int allSlots[1..nbGames] in 1..nbGames;
055: dvar boolean playHome[1..n][1..nbWeeks];
056: dvar int allGames[1..nbGames] = all[1..nbGames](w in 1..nbWeeks, g in 1..nbGamesPerWeek) games[w][g];
057: dvar int teamBreaks[1..n] in 0..nbWeeks div 2;
058: 
059: //
060: // For each play slot, set up correspondance between game id,
061: // home team, and away team
062: tuple PlaySlotTuple {
063:    int home;
064:    int away;
065:    key int gameId;
066: };
067: 
068: {PlaySlotTuple} playSlots = {<h, a, (h-1) * (n-1) + a - (a > h)> | h, a in 1..n : a != h};
069: 
070: 
071: execute {
072:     var f = cp.factory;
073:     var phase = f.searchPhase(allGames, 
074:                               f.selectSmallest(f.varIndex(allGames)), 
075:                               f.selectRandomValue());
076:     cp.setSearchPhases(phase);
077:     cp.param.timeLimit=60;
078:     cp.param.logPeriod=10000;
079:     cp.param.DefaultInferenceLevel="Extended";
080: }
081: 
082: 
083: //
084: // Objective: minimize the number of `breaks'.  A break is
085: //            two consecutive home or away matches for a
086: //            particular team
087: dexpr int breakCount = sum(t in 1..n) teamBreaks[t];
088: 
089: 
090: minimize breakCount;
091: // minimize sum(t in 1..n) teamBreaks[t];
092: subject to {
093: 
094:    forall(w in 1..nbWeeks, g in 1..nbGamesPerWeek)
095:      allowedAssignments(playSlots, home[w][g], away[w][g], games[w][g]);
096: 
097: 
098:    //
099:    // All teams play each week
100:    //
101:    forall(w in 1..nbWeeks) {
102: 
103:      allDifferent(append(all(g in 1..nbGamesPerWeek) home[w][g], 
104:                          all(g in 1..nbGamesPerWeek) away[w][g])); 
105:                 
106:    }
107:                          
108: 
109:     //
110:     // Dual representation: for each game id, the play slot is maintained
111:     //
112:     inverse(all [1..nbGames](w in 1..nbWeeks, g in 1..nbGamesPerWeek) games[w][g], allSlots); 
113:     forall(g in 1..nbGames)
114:       weekOfGame[g] == ((allSlots[g]-1) div nbGamesPerWeek) + 1;
115: 
116: 
117:     //
118:     // Two half schedules.  Cannot play the same pair twice in the same half.
119:     // Plus, impose a minimum number of weeks between two games involving
120:     // the same teams (up to six weeks)
121:     //
122:     forall (<i,j,g1> in playSlots, <j,i,g2> in playSlots  : i < j) {
123:        (weekOfGame[g1] >= mid) == (weekOfGame[g2] < mid); 
124:        if (overlap != 0)
125:           abs(weekOfGame[g1] - weekOfGame[g2]) >= overlap;
126:     }
127: 
128: 
129:                
130:     //
131:     // Can't have three homes or three aways in a row.
132:     //
133:     forall (t in 1..n, w in 1..nbWeeks) {
134:        playHome[t][w] == count(all(g in 1..nbGamesPerWeek) home[w][g], t);
135:     }
136: 
137:     forall (t in 1..n, w in 1..nbWeeks - 2) {
138:        1 <= sum(k in w..w+2) playHome[t][k] <= 2;
139:     }
140:   
141:     //
142:     // If we start the season home, we finish away and vice versa.
143:     //
144:     forall(t in 1..n)
145:        teamBreaks[t] == sum(w in 2..nbWeeks) (playHome[t][w-1] == playHome[t][w]);
146:        
147:     forall (t in 1..n)
148:       playHome[t][1] != playHome[t][nbWeeks];
149: 
150: 
151:     //
152:     // Catalyzing constraints
153:     //
154: 
155:     // Each team plays home the same number of times as away
156:     forall (t in 1..n) {
157:        sum(w in 1..nbWeeks) playHome[t][w] == nbWeeks div 2;
158:     }
159: 
160:     // Breaks must be even for each team
161:     forall(t in 1..n)
162:        teamBreaks[t] % 2 == 0;
163: 
164: 
165: 
166:     //    
167:     // Symmetry breaking constraints
168:     // 
169:     // Teams are interchangeable.  Fix first week.
170:     // Also breaks reflection symmetry of the whole schedule.
171:     forall (g in 1..nbGamesPerWeek) {
172:        home[1][g] == g*2 - 1;
173:        away[1][g] == g*2;
174:     }
175: 
176: 
177:     // Order of games in each week in arbitrary.
178:     // Break symmetry by forcing an order.
179:     forall (w in 1..nbWeeks)
180:       forall(g in 2..nbGamesPerWeek)
181:         games[w][g] > games[w][g-1];
182:        
183: 
184: }
185: 
186: int oponent[1..n][1..nbWeeks];
187: 
188: int breaks = sum(t in 1..n) teamBreaks[t];
189: 
190: 
191: 
192: execute {
193:    
194:       writeln("Solution at " + breaks);
195:       for (var j= 1; j <= nbWeeks; j++) {
196:          write("Week " + j + ": ");
197:          if (j < 10) write (" ");
198:          for (var i = 1; i <= nbGamesPerWeek; i++) {
199:             if (home[j][i] >= 10) 
200:               write(home[j][i]);
201:             else
202:               write(" " + home[j][i]);
203:             write("-");
204:             if (away[j][i] >= 10)
205:               write(away[j][i]);
206:             else
207:               write(away[j][i] + " ");
208:             write(" ");      
209:          }
210:          writeln();
211:       }
212:       writeln("Team schedules");
213:       for (i = 1; i <= n; i++) {
214:          write("T " + i + ":  ");
215:          var prev = -1;
216:          var brks = 0;
217:          for (j = 1; j <= nbWeeks; j++) {
218:             for (var k = 1; k <= nbGamesPerWeek; k++) {
219:                if (home[j][k] == i) {
220:                   oponent[i][j] = away[j][k];
221:                   if (away[j][k] >= 10)
222:                     write(away[j][k] + "H ")
223:                   else
224:                     write(" " + away[j][k] + "H ");
225:                   brks += (prev == 0);
226:                   prev = 0;    
227:                }
228:                if (away[j][k] == i) {
229:                   oponent[i][j] = home[j][k];
230:                   if (home[j][k] >= 10)
231:                     write(home[j][k] + "A ");
232:                   else
233:                     write(" " + home[j][k] + "A ");
234:                   brks += (prev == 1);
235:                   prev = 1;
236:                }
237:             }
238:          }
239:          writeln("  " + brks + " breaks");
240:       }
241:       writeln();
242: }

258: {solution2DimT} homeReport = {<i,j, home[i][j]> | i in 1..nbWeeks, j in 1..nbGamesPerWeek};
259: {solution1DimT} awayReport = {<i, away[i][j]> | i in 1..nbWeeks, j in 1..nbGamesPerWeek};
260: {solution1DimT} weekOfGameReport = {<i, weekOfGame[i]> | i in 1..nbGames};
261: {solution1DimT} allSlotsReport = {<i, allSlots[i]> | i in 1..nbGames};
262: {solution2DimT} playHomeReport = {<i,j, playHome[i][j]> | i in 1..n, j in 1..nbWeeks};
263: {solution1DimT} allGamesReport = {<i, allGames[i]> | i in 1..nbGames};
264: {solution1DimT} teamBreaksReport = {<i, teamBreaks[i]> | i in 1..n};
