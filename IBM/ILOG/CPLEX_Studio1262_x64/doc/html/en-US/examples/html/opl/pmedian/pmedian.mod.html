12: //Data
13: int P = ...;
14: {string} Customers = ...;
15: {string} Warehouses = ...;
16: int Demand[Customers] = ...;
17: float Distance[Customers][Warehouses] = ...;
18: 
19: //Variables
20: dvar boolean OpenWarehouse[Warehouses];
21: dvar boolean ShipToCustomer[Customers][Warehouses];
22: 
23: //Objective
24: minimize 
25:   sum( c in Customers , w in Warehouses ) 
26:     Demand[c]*Distance[c][w]*ShipToCustomer[c][w];
27: 
28: //Constraints
29: subject to {
30:   forall( c in Customers )
31:     ctShip:
32:       sum( w in Warehouses ) 
33:         ShipToCustomer[c][w] == 1;
34: 
35:   ctOpen:
36:     sum( w in Warehouses ) 
37:       OpenWarehouse[w] == P;
38: 
39:   forall( c in Customers , w in Warehouses )
40:     ctShipOpen:
41:       ShipToCustomer[c][w] <= OpenWarehouse[w];
42: }
43: 
44: tuple ShipToCustomerSolutionT{ 
45:         string Customers; 
46:         string Warehouses; 
47:         int value; 
48: };
49: {ShipToCustomerSolutionT} ShipToCustomerSolution = {<i0,i1,ShipToCustomer[i0][i1]> | i0 in Customers,i1 in Warehouses};
50: tuple OpenWarehouseSolutionT{ 
51:         string Warehouses; 
52:         int value; 
53: };
54: {OpenWarehouseSolutionT} OpenWarehouseSolution = {<i0,OpenWarehouse[i0]> | i0 in Warehouses};
