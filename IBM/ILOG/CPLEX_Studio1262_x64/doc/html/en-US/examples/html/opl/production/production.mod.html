12: {string} Products = ...;
13: {string} Resources = ...;
14: 
15: float Consumption[Products][Resources] = ...;
16: float Capacity[Resources] = ...;
17: float Demand[Products] = ...;
18: float InsideCost[Products] = ...;
19: float OutsideCost[Products]  = ...;
20: 
21: dvar float+ Inside[Products];
22: dvar float+ Outside[Products];
23: 
24: minimize
25:   sum( p in Products ) 
26:     ( InsideCost[p] * Inside[p] + OutsideCost[p] * Outside[p] );
27:    
28: subject to {
29:   forall( r in Resources )
30:     ctCapacity: 
31:       sum( p in Products ) 
32:         Consumption[p][r] * Inside[p] <= Capacity[r];
33: 
34:   forall(p in Products)
35:     ctDemand:
36:       Inside[p] + Outside[p] >= Demand[p];
37: }
