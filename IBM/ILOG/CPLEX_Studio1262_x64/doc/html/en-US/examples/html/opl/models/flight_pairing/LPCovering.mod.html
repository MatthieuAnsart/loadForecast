12: ///
13: include "FlightPairing_data.mod";
14: include "FlightCoverDual_data.mod";
15: include "PairingGeneration_data.mod";
16: 
17: /// externals specific to this LP model...
18: {TPairingCost} pairingCosts = ...;
19: {TPairingLeg}  pairingLegs  = ...;
20: 
21: 
22: /// internal
23: int NbFlights = card(flights);
24: range FlightIndexRange = 1..NbFlights;
25: 
26: /// locals
27: {int} covered[ p in pairingCosts ] = { fr | <p.id, l, fr> in pairingLegs };
28: 
29: dvar float usageVars[ pairingCosts] in 0..1;
30: 
31: dexpr float totalUsageCost = sum(pc in pairingCosts) (usageVars[pc] * pc.cost);
32: // symmetry breaking?? minimize nb of pairings, costs being equal.
33: //dexpr float totalUsed =  sum(pc in pairingCosts) (usageVars[pc] );
34: 
35: constraint flight_cover_cts[ 1..NbFlights ];
36: 
37: minimize totalUsageCost;
38: 
39: subject to { 
40:   forall (f in 1..NbFlights) {
41:    flight_cover_cts[f]:
42:     sum(pc in pairingCosts: f in covered[pc] ) usageVars[pc] >= 1;  
43:   }
44: }
45: 
46: /// ---
47: ///  post-processing: save dual values per flight
48: /// ---
49: 
50: {TFlightCoverDual} coverCtDuals;
51: float dualHash;
52: 
53: execute DUMP_DUALS{
54: 
55:   includeScript("helpers.js");
56:   
57:   var DUAL_EPS = 1e-5;
58:   thisOplModel.settings.displayPrecision = 8;
59:   
60:   thisOplModel.dualHash = 0;
61:   for ( var f in FlightIndexRange) {
62:     coverCtDuals.add(f, flight_cover_cts[f].dual <= DUAL_EPS ? 0 : flight_cover_cts[f].dual);
63:     thisOplModel.dualHash += flight_cover_cts[f].dual <= DUAL_EPS ? 0 : flight_cover_cts[f].dual;
64:   }//for
65:   
66:   /// debug: write duals as a DAT structure
67:   if ( thisOplModel.PairingGenerationData.PrintFiles) {
68:      print_dual_values("sample_cover_duals.dat", coverCtDuals);
69:    }     
70: }
