12: range r=1..3;
13: dvar boolean cell[r][r][r];
14: 
15: tuple t
16: {
17:    int x1;
18:    int x2;
19:    int x3;
20:    int y1;
21:    int y2;
22:    int y3;
23:    int z1;
24:    int z2;
25:    int z3;   
26: }
27: 
28: {t} lines={<x1,y1,z1,x2,y2,z2,x3,y3,z3> | x1,x2,x3,y1,y2,y3,z1,z2,z3 in r 
29: : (x2-x1==x3-x2) && (y2-y1==y3-y2) && (z2-z1==z3-z2) && (x1*9+y1*3+z1<x3*9+y3*3+z3) };
30: 
31: assert card(lines)==49;
32: 
33: dvar boolean monocolor[lines];
34: 
35: // minimize the number of mono color lines
36: minimize sum(l in lines) monocolor[l];
37: 
38: subject to
39: {
40:   // a line is mono color implies all the cells in the line have the same color 
41:   forall(<x1,y1,z1,x2,y2,z2,x3,y3,z3> in lines) 
42:     (monocolor[<x1,y1,z1,x2,y2,z2,x3,y3,z3>]==0) => 
43:     (cell[x1][y1][z1]+cell[x2][y2][z2]+cell[x3][y3][z3]>=1 && 
44:     cell[x1][y1][z1]+cell[x2][y2][z2]+cell[x3][y3][z3]<=2);
45:   
46:   // 14 cells of the same color
47:   sum(i,j,k in r) cell[i][j][k]==14; 
48: }
49: 
50: tuple monoT{
51:         t id;
52:         int value;
53: }
54: {monoT} monocolorSolution = {<t, monocolor[t]> | t in lines};
55: tuple cellsT{
56:         int a;
57:         int b;
58:         int c;
59:         int value;
60: }
61: {cellsT} cellSolution = {<i,j,k,cell[i][j][k]> | i,j,k in r};
