12: {string} Machines = ...;
13: {string} Products = ...;
14: {string} Resources = ...;
15: 
16: int Capacity[Resources] = ...;
17: int MaxProduction = max(r in Resources) Capacity[r];
18: int RentingCost[Machines] = ...;
19: tuple productType {
20:    int profit;
21:    {string} machines;
22:    int use[Resources];
23: }
24: productType Product[Products] = ...;
25: 
26: dvar boolean Rent[Machines];
27: dvar int Produce[Products] in 0..MaxProduction;
28: 
29: constraint ctMaxProd[Products][Machines];
30: 
31: maximize 
32:   sum( p in Products ) 
33:     Product[p].profit * Produce[p] -
34:   sum( m in Machines ) 
35:     RentingCost[m] * Rent[m];
36:       
37: subject to {
38:   forall( r in Resources )
39:     ctCapacity:
40:       sum( p in Products ) 
41:         Product[p].use[r] * Produce[p] <= Capacity[r];
42:     forall( p in Products , m in Product[p].machines )
43:       ctMaxProd[p][m]:
44:         Produce[p] <= MaxProduction * Rent[m];
45: }
46: 
47: 
48: tuple ProduceSolutionT{ 
49:         string Products; 
50:         int value; 
51: };
52: {ProduceSolutionT} ProduceSolution = {<i0,Produce[i0]> | i0 in Products};
53: tuple RentSolutionT{ 
54:         string Machines; 
55:         int value; 
56: };
57: {RentSolutionT} RentSolution = {<i0,Rent[i0]> | i0 in Machines};
