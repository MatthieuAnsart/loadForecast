025:   
026: //**************************** Data **************************************
027: int Totalshifts = ...; // Total # of shifts in a day
028: int Nbshifts = ...;    // Total # shifts a person should work in a day
029: range Shifts = 1..Totalshifts; 
030: 
031: {string} Skills = ...;     // Type of skills required
032: {string} Weekdays = ...;   // Set of work days 
033: {string} People = ...;     // Set of personnel
034: 
035: int Req[Weekdays][Shifts][Skills] = ...;  // # of persons of a skill type required in each shifts
036:                   
037: // Data Structure
038: tuple shift {
039:   key string p;
040:   string w;
041: }
042: 
043: tuple PSkill {
044:   key string p;
045:   {string} s;
046: }
047: 
048: {PSkill} PeopleSkills = ...; //  List of skills that a person has 
049: {shift}  Unavailable = ...;  // Unavailability of each person
050: 
051: int Penalty = card(Weekdays)*Nbshifts+1; // Penalty for an unfilled slot
052: 
053: //********************************* Variables **********************************
054:         
055: dvar boolean Assign[Weekdays][Shifts][People][Skills];   // Indicates a shift assignment
056: dvar int Unfilled[Weekdays][Shifts][Skills] in 0..maxint;  // # of skilled persons unfilled in a shift
057: dvar int Pmin           in 0..card(Weekdays)*Nbshifts;     // Minimal # of shifts assigned 
058: dvar int Pmax           in 0..card(Weekdays)*Nbshifts;     // Maximal # of shifts assigned
059: dvar boolean Start[Weekdays][Shifts][People];             // People who start working in a shift
060: dvar int Personnel[Weekdays][Shifts] in 0..card(People);     // # of avail personnel in a shift
061: dvar boolean Avail[Weekdays][Shifts][People];            // Indicates the availability of a person in a shift
062: 
063: 
064: // the total # of slots unfilled in a week
065: dexpr int TotUnfilled  =
066:   sum(w in Weekdays, s in Shifts, j in Skills) Unfilled[w][s][j];
067: 
068: /************************************* Model *********************************/
069: 
070: minimize TotUnfilled*Penalty + (Pmax-Pmin);
071: // Note:  Since the penalty is higher than the maximal 
072: // possible difference in the # of shifts assigned to the workers,
073: // the schedule always fills the demand first, and then balances the work load.
074: 
075: subject to {
076:    
077:   // The number of available personnel in a shift = 
078:   // the sum of the number of starting personnel in pervious "nbshifts" shifts.
079:   forall(w in Weekdays, s in Shifts)   
080:     Personnel[w][s] == sum(i in Shifts: s-Nbshifts+1 <= i <= s, j in People) Start[w][i][j];
081:       
082:   // The number of personnel in a shift = total # of available persons. 
083:   forall(w in Weekdays, s in Shifts)  
084:     Personnel[w][s] == sum(j in People) Avail[w][s][j];
085:    
086:   // the total # of person working in a shift <= the # of personnel in the shift.
087: 
088:   forall(w in Weekdays, s in Shifts)  
089:     ctPersonnel:       
090:       sum(p in People, j in Skills) Assign[w][s][p][j] <= Personnel[w][s];
091:       
092:   forall(w in Weekdays, s in Shifts,p in People) {
093:     // If a person start in shift k, she will be available during next nbshifts shifts. 
094:     forall(k in Shifts: s-Nbshifts+1 <= k <= s) 
095:       Avail[w][s][p] >= Start[w][k][p];
096:     // If a person has started, she can't "start" again in any of the next nbshifts shifts. 
097:     forall(k in Shifts: s+1 <= k <= s+Nbshifts-1) 
098:       1-Start[w][s][p] >= Start[w][k][p];
099:   }
100:    
101:   // If a person is available, he must be assigned a job       
102:   forall(w in Weekdays, s in Shifts, p in People)
103:     sum(j in Skills) Assign[w][s][p][j] == Avail[w][s][p];       
104:     
105:   // In each shift,  # of people assigned to a type of task +
106:   //   unfilled slot >= # of people of that type required
107:   forall(w in Weekdays, s in Shifts, j in Skills)
108:     ctUnfilled:    
109:       sum(p in People) 
110:         Assign[w][s][p][j] + Unfilled[w][s][j] >= Req[w][s][j];
111:    
112:   forall(w in Weekdays, s in Shifts, t in PeopleSkills, k in Skills: k not in t.s)
113:     Assign[w][s][t.p][k] == 0;
114:         
115:   // Everyone has to work no longer than 8 hours (2 shifts in this case)
116:   forall(w in Weekdays, p in People) 
117:     ctShifts:      
118:       sum(s in Shifts, j in Skills) 
119:         Assign[w][s][p][j] <= Nbshifts;
120:         
121:   // Unavailable person cannot be assigned on that day
122:   forall(<p,w> in Unavailable, s in Shifts, j in Skills)
123:     Assign[w][s][p][j] == 0;
124: 
125:   // Each person can take only one task in each shift     
126:   forall(w in Weekdays, s in Shifts, p in People) 
127:     ctTasks:      
128:       sum(j in Skills) 
129:         Assign[w][s][p][j] <= 1;
130:  
131:   // If a person is on night shift, he cannot be assigned to the morning
132:   //  shift the next day
133:   forall(p in People, k in Skills) {   
134:     Assign["Tue"][1][p][k] <= 1 - sum(j in Skills) Assign["Mon"][Totalshifts][p][j];
135:     Assign["Wed"][1][p][k] <= 1 - sum(j in Skills) Assign["Tue"][Totalshifts][p][j];
136:     Assign["Thu"][1][p][k] <= 1 - sum(j in Skills) Assign["Wed"][Totalshifts][p][j];
137:     Assign["Fri"][1][p][k] <= 1 - sum(j in Skills) Assign["Thu"][Totalshifts][p][j];
138:   }
139:   
140:   //Each shift at least has a cooker, a cleaner and a cashier
141:   forall(w in Weekdays, s in Shifts, j in Skills) 
142:     ctSkills:
143:       sum(p in People) Assign[w][s][p][j] >= 1;     
144:    
145:   // The workload of a person is bounded by pmin and pmax 
146:   forall(p in People)
147:     Pmin <= sum(w in Weekdays, s in Shifts, j in Skills) Assign[w][s][p][j];  
148:   forall(p in People)
149:     sum(w in Weekdays, s in Shifts, j in Skills) Assign[w][s][p][j] <= Pmax;  
150:        
151: }
152: 
153: 
154: tuple UnfilledSolutionT{ 
155:         string Weekdays; 
156:         int Shifts; 
157:         string Skills; 
158:         int value; 
159: };
160: {UnfilledSolutionT} UnfilledSolution = {<i0,i1,i2,Unfilled[i0][i1][i2]> | i0 in Weekdays,i1 in Shifts,i2 in Skills};
161: tuple PersonnelSolutionT{ 
162:         string Weekdays; 
163:         int Shifts; 
164:         int value; 
165: };
166: {PersonnelSolutionT} PersonnelSolution = {<i0,i1,Personnel[i0][i1]> | i0 in Weekdays,i1 in Shifts};
167: tuple StartSolutionT{ 
168:         string Weekdays; 
169:         int Shifts; 
170:         string People; 
171:         int value; 
172: };
173: {StartSolutionT} StartSolution = {<i0,i1,i2,Start[i0][i1][i2]> | i0 in Weekdays,i1 in Shifts,i2 in People};
174: tuple AvailSolutionT{ 
175:         string Weekdays; 
176:         int Shifts; 
177:         string People; 
178:         int value; 
179: };
180: {AvailSolutionT} AvailSolution = {<i0,i1,i2,Avail[i0][i1][i2]> | i0 in Weekdays,i1 in Shifts,i2 in People};
181: tuple AssignSolutionT{ 
182:         string Weekdays; 
183:         int Shifts; 
184:         string People; 
185:         string Skills; 
186:         int value; 
187: };
188: {AssignSolutionT} AssignSolution = {<i0,i1,i2,i3,Assign[i0][i1][i2][i3]> | i0 in Weekdays,i1 in Shifts,i2 in People,i3 in Skills};
