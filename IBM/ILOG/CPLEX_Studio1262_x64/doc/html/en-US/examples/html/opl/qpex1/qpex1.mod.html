12: /* QP: Quadratic Objective
13:  *
14:  * Example from CPLEX : iloqpex1.cpp 
15:  */
16: range R = 0..2;
17: dvar float x[R] in 0..40;
18: 
19: 
20: maximize
21:   x[0] + 2 * x[1] + 3 * x[2]
22:   - 0.5 * 
23:     ( 33*x[0]^2 + 22*x[1]^2 + 11*x[2]^2 
24:     - 12*x[0]*x[1] - 23*x[1]*x[2] );
25: 
26: subject to {
27:   ct1:  - x[0] +     x[1] + x[2] <= 20;
28:   ct2:    x[0] - 3 * x[1] + x[2] <= 30;
29: }
30: 
31: tuple xSolutionT{ 
32:         int R; 
33:         float value; 
34: };
35: {xSolutionT} xSolution = {<i0,x[i0]> | i0 in R};
